9use std::ptr;
10
11/// Partial is a partially written contiguous slice of data;
12/// it is the owner of the elements, but not the allocation,
13/// and will drop the elements on drop.
14#[must_use]
15pub(crate) struct Partial<T>
16{
17    /// Data pointer
18    ptr: *mut T,
19    /// Current length
20    pub(crate) len: usize,
21}
22
23impl<T> Partial<T>
24{
25    /// Create an empty partial for this data pointer
26    ///
27    /// ## Safety
28    ///
29    /// Unless ownership is released, the Partial acts as an owner of the slice of data (not the
30    /// allocation); and will free the elements on drop; the pointer must be dereferenceable and
31    /// the `len` elements following it valid.
32    ///
33    /// The Partial has an accessible length field which must only be modified in trusted code.
34    pub(crate) unsafe fn new(ptr: *mut T) -> Self
35    {
36        Self { ptr, len: 0 }
37    }
38
39    #[cfg(feature = "rayon")]
40    pub(crate) fn stub() -> Self
41    {
42        Self {
43            len: 0,
44            ptr: ptr::null_mut(),
45        }
46    }
47
48    #[cfg(feature = "rayon")]
49    pub(crate) fn is_stub(&self) -> bool
50    {
51        self.ptr.is_null()
52    }
53
54    /// Release Partial's ownership of the written elements, and return the current length
55    pub(crate) fn release_ownership(mut self) -> usize
56    {
57        let ret = self.len;
58        self.len = 0;
59        ret
60    }
61
62    #[cfg(feature = "rayon")]
63    /// Merge if they are in order (left to right) and contiguous.
64    /// Skips merge if T does not need drop.
65    pub(crate) fn try_merge(mut left: Self, right: Self) -> Self
66    {
67        if !std::mem::needs_drop::<T>() {
68            return left;
69        }
70        // Merge the partial collect results; the final result will be a slice that
71        // covers the whole output.
72        if left.is_stub() {
73            right
74        } else if left.ptr.wrapping_add(left.len) == right.ptr {
75            left.len += right.release_ownership();
76            left
77        } else {
78            // failure to merge; this is a bug in collect, so we will never reach this
79            debug_assert!(false, "Partial: failure to merge left and right parts");
80            left
81        }
82    }
83}
84
85unsafe impl<T> Send for Partial<T> where T: Send {}
86
87impl<T> Drop for Partial<T>
88{
89    fn drop(&mut self)
90    {
91        if !self.ptr.is_null() {
92            unsafe {
93                ptr::drop_in_place(alloc::slice::from_raw_parts_mut(self.ptr, self.len));
94            }
95        }
96    }
97}
