8/// Partitions elements into groups of clusters along a specific axis.
9pub struct ClusterGroupIterator<T: RTreeObject> {
10    remaining: Vec<T>,
11    slab_size: usize,
12    pub cluster_dimension: usize,
13}
14
15impl<T: RTreeObject> ClusterGroupIterator<T> {
16    pub fn new(
17        elements: Vec<T>,
18        number_of_clusters_on_axis: usize,
19        cluster_dimension: usize,
20    ) -> Self {
21        let slab_size = div_up(elements.len(), number_of_clusters_on_axis);
22        ClusterGroupIterator {
23            remaining: elements,
24            slab_size,
25            cluster_dimension,
26        }
27    }
28}
29
30impl<T: RTreeObject> Iterator for ClusterGroupIterator<T> {
31    type Item = Vec<T>;
32
33    fn next(&mut self) -> Option<Self::Item> {
34        match self.remaining.len() {
35            0 => None,
36            len if len <= self.slab_size => ::core::mem::take(&mut self.remaining).into(),
37            _ => {
38                let slab_axis = self.cluster_dimension;
39                T::Envelope::partition_envelopes(slab_axis, &mut self.remaining, self.slab_size);
40                let off_split = self.remaining.split_off(self.slab_size);
41                ::core::mem::replace(&mut self.remaining, off_split).into()
42            }
43        }
44    }
45}

47/// Calculates the desired number of clusters on any axis
48///
49/// A 'cluster' refers to a set of elements that will finally form an rtree node.
50pub fn calculate_number_of_clusters_on_axis<T, Params>(number_of_elements: usize) -> usize
51where
52    T: RTreeObject,
53    Params: RTreeParams,
54{
55    let max_size = Params::MAX_SIZE as f32;
56    // The depth of the resulting tree, assuming all leaf nodes will be filled up to MAX_SIZE
57    let depth = (number_of_elements as f32).log(max_size).ceil() as usize;
58    // The number of elements each subtree will hold
59    let n_subtree = (max_size as f32).powi(depth as i32 - 1);
60    // How many clusters will this node contain
61    let number_of_clusters = (number_of_elements as f32 / n_subtree).ceil();
62
63    let max_dimension = <T::Envelope as Envelope>::Point::DIMENSIONS as f32;
64    // Try to split all clusters among all dimensions as evenly as possible by taking the nth root.
65    number_of_clusters.powf(1. / max_dimension).ceil() as usize
66}
67
68fn div_up(dividend: usize, divisor: usize) -> usize {
69    (dividend + divisor - 1) / divisor
70}

72#[cfg(test)]
73mod test {
74    use super::ClusterGroupIterator;
75
76    #[test]
77    fn test_cluster_group_iterator() {
78        const SIZE: usize = 374;
79        const NUMBER_OF_CLUSTERS_ON_AXIS: usize = 5;
80        let elements: Vec<_> = (0..SIZE as i32).map(|i| [-i, -i]).collect();
81        let slab_size = (elements.len()) / NUMBER_OF_CLUSTERS_ON_AXIS + 1;
82        let slabs: Vec<_> =
83            ClusterGroupIterator::new(elements, NUMBER_OF_CLUSTERS_ON_AXIS, 0).collect();
84        assert_eq!(slabs.len(), NUMBER_OF_CLUSTERS_ON_AXIS);
85        for slab in &slabs[0..slabs.len() - 1] {
86            assert_eq!(slab.len(), slab_size);
87        }
88        let mut total_size = 0;
89        let mut max_element_for_last_slab = i32::min_value();
90        for slab in &slabs {
91            total_size += slab.len();
92            let current_max = slab.iter().max_by_key(|point| point[0]).unwrap();
93            assert!(current_max[0] > max_element_for_last_slab);
94            max_element_for_last_slab = current_max[0];
95        }
96        assert_eq!(total_size, SIZE);
97    }
