1use std::borrow::Borrow;
2use std::fmt::Display;
3
4use crate::geometry::Transformation;
5use ordered_float::NotNan;
6
7#[derive(Clone, Debug, PartialEq, Eq, Hash, Copy, Default)]
8/// [Proper rigid transformation](https://en.wikipedia.org/wiki/Rigid_transformation),
9/// decomposed into a rotation followed by a translation.
10pub struct DTransformation {
11    /// The rotation in radians
12    pub rotation: NotNan<f32>,
13    /// The translation in the x and y-axis
14    pub translation: (NotNan<f32>, NotNan<f32>),
15}
16
17impl DTransformation {
18    pub fn new(rotation: f32, translation: (f32, f32)) -> Self {
19        Self {
20            rotation: NotNan::new(rotation).expect("rotation is NaN"),
21            translation: (
22                NotNan::new(translation.0).expect("translation.0 is NaN"),
23                NotNan::new(translation.1).expect("translation.1 is NaN"),
24            ),
25        }
26    }
27
28    pub const fn empty() -> Self {
29        const _0: NotNan<f32> = unsafe { NotNan::new_unchecked(0.0) };
30        Self {
31            rotation: _0,
32            translation: (_0, _0),
33        }
34    }
35
36    pub fn rotation(&self) -> f32 {
37        self.rotation.into()
38    }
39
40    pub fn translation(&self) -> (f32, f32) {
41        (self.translation.0.into(), self.translation.1.into())
42    }
43
44    pub fn compose(&self) -> Transformation {
45        self.into()
46    }
47}
48
49impl<T> From<T> for DTransformation
50where
51    T: Borrow<Transformation>,
52{
53    fn from(t: T) -> Self {
54        t.borrow().decompose()
55    }
56}
57
58impl Display for DTransformation {
59    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
60        write!(
61            f,
62            "r: {:.3}°, t: ({:.3}, {:.3})",
63            self.rotation.to_degrees(),
64            self.translation.0.into_inner(),
65            self.translation.1.into_inner()
66        )
67    }
68}
