20/// The Collision Detection Engine (CDE).
21/// [`Hazard`]s can be (de)registered and collision queries can be performed.

33    pub fn new(bbox: Rect, static_hazards: Vec<Hazard>, config: CDEConfig) -> CDEngine {
34        let mut qt_root = QTNode::new(config.quadtree_depth, bbox);
35
36        for haz in static_hazards.iter() {
37            let qt_haz = QTHazard::from_qt_root(qt_root.bbox, haz);
38            qt_root.register_hazard(qt_haz);
39        }
40
41        CDEngine {
42            quadtree: qt_root,
43            static_hazards,
44            dynamic_hazards: vec![],
45            config,
46            bbox,
47            uncommitted_deregisters: vec![],
48        }
49    }

51    /// Registers a new hazard in the CDE.
52    pub fn register_hazard(&mut self, hazard: Hazard) {
53        debug_assert!(
54            !self
55                .dynamic_hazards
56                .iter()
57                .any(|h| h.entity == hazard.entity),
58            "Hazard already registered"
59        );
60        let hazard_in_uncommitted_deregs = self
61            .uncommitted_deregisters
62            .iter()
63            .position(|h| h.entity == hazard.entity);
64
65        let hazard = match hazard_in_uncommitted_deregs {
66            Some(index) => {
67                let unc_hazard = self.uncommitted_deregisters.swap_remove(index);
68                self.quadtree.activate_hazard(unc_hazard.entity);
69                unc_hazard
70            }
71            None => {
72                let qt_haz = QTHazard::from_qt_root(self.bbox, &hazard);
73                self.quadtree.register_hazard(qt_haz);
74                hazard
75            }
76        };
77        self.dynamic_hazards.push(hazard);
78
79        debug_assert!(assertions::qt_contains_no_dangling_hazards(self));
80    }

82    /// Removes a hazard from the CDE.
83    /// If `commit_instant` the deregistration is fully executed immediately.
84    /// If not, the deregistration causes the hazard to be deactivated in the quadtree and
85    /// the hazard_proximity_grid to become dirty (and therefore inaccessible).
86    /// <br>
87    /// Can be beneficial not to `commit_instant` if multiple hazards are to be deregistered, or if the chance of
88    /// restoring from a snapshot with the hazard present is high.
89    pub fn deregister_hazard(&mut self, hazard_entity: HazardEntity, commit_instant: bool) {
90        let haz_index = self
91            .dynamic_hazards
92            .iter()
93            .position(|h| h.entity == hazard_entity)
94            .expect("Hazard not found");
95
96        let hazard = self.dynamic_hazards.swap_remove(haz_index);
97
98        match commit_instant {
99            true => self.quadtree.deregister_hazard(hazard_entity),
100            false => {
101                self.quadtree.deactivate_hazard(hazard_entity);
102                self.uncommitted_deregisters.push(hazard);
103            }
104        }
105        debug_assert!(assertions::qt_contains_no_dangling_hazards(self));
106    }

108    pub fn create_snapshot(&mut self) -> CDESnapshot {
109        self.commit_deregisters();
110        CDESnapshot {
111            dynamic_hazards: self.dynamic_hazards.clone(),
112        }

115    /// Restores the CDE to a previous state, as described by the snapshot.
116    pub fn restore(&mut self, snapshot: &CDESnapshot) {
117        //Quadtree
118        let mut hazards_to_remove = self.dynamic_hazards.iter().map(|h| h.entity).collect_vec();
119        debug_assert!(hazards_to_remove.len() == self.dynamic_hazards.len());
120        let mut hazards_to_add = vec![];
121
122        for hazard in snapshot.dynamic_hazards.iter() {
123            let hazard_already_present = hazards_to_remove.iter().position(|h| h == &hazard.entity);
124            if let Some(idx) = hazard_already_present {
125                //the hazard is already present in the CDE, remove it from the hazards to remove
126                hazards_to_remove.swap_remove(idx);
127            } else {
128                //the hazard is not present in the CDE, add it to the list of hazards to add
129                hazards_to_add.push(hazard.clone());
130            }
131        }
132
133        //Hazards currently registered in the CDE, but not in the snapshot
134        for haz_entity in hazards_to_remove.iter() {
135            let haz_index = self
136                .dynamic_hazards
137                .iter()
138                .position(|h| &h.entity == haz_entity)
139                .expect("Hazard not found");
140            self.dynamic_hazards.swap_remove(haz_index);
141            self.quadtree.deregister_hazard(*haz_entity);
142        }
143
144        //Some of the uncommitted deregisters might be in present in snapshot, if so we can just reactivate them
145        for unc_haz in self.uncommitted_deregisters.drain(..) {
146            if let Some(pos) = hazards_to_add
147                .iter()
148                .position(|h| h.entity == unc_haz.entity)
149            {
150                //the uncommitted removed hazard needs to be activated again
151                self.quadtree.activate_hazard(unc_haz.entity);
152                self.dynamic_hazards.push(unc_haz);
153                hazards_to_add.swap_remove(pos);
154            } else {
155                //uncommitted deregister is not preset in the snapshot, delete it from the quadtree
156                self.quadtree.deregister_hazard(unc_haz.entity);
157            }
158        }
159
160        for hazard in hazards_to_add {
161            let qt_haz = QTHazard::from_qt_root(self.bbox, &hazard);
162            self.quadtree.register_hazard(qt_haz);
163            self.dynamic_hazards.push(hazard);
164        }
165
166        debug_assert!(self.dynamic_hazards.len() == snapshot.dynamic_hazards.len());
167    }

169    /// Commits all pending deregisters by actually removing them from the quadtree
170    pub fn commit_deregisters(&mut self) {
171        for uncommitted_hazard in self.uncommitted_deregisters.drain(..) {
172            self.quadtree.deregister_hazard(uncommitted_hazard.entity);
173        }
174    }

213    ///Checks whether a reference simple polygon, with a transformation applies, collides with any of the hazards.
214    ///The check is first done on the surrogate, then with the actual shape.
215    ///A buffer shape is used as a temporary storage for the transformed shape.
216    /// # Arguments
217    /// * `reference_shape` - The shape to be checked for collisions
218    /// * `transform` - The transformation to be applied to the reference shape
219    /// * `buffer_shape` - A temporary storage for the transformed shape
220    /// * `filter` - Hazard filter to be applied
221    pub fn surrogate_or_poly_collides(
222        &self,
223        reference_shape: &SPolygon,
224        transform: &Transformation,
225        buffer_shape: &mut SPolygon,
226        filter: &impl HazardFilter,
227    ) -> bool {
228        //Begin with checking the surrogate for collisions
229        match self.surrogate_collides(reference_shape.surrogate(), transform, filter) {
230            true => true,
231            false => {
232                //Transform the reference_shape and store the result in the buffer_shape
233                buffer_shape.transform_from(reference_shape, transform);
234                self.poly_collides(buffer_shape, filter)
235            }
236        }
237    }

239    ///Checks whether a simple polygon collides with any of the (relevant) hazards
240    /// # Arguments
241    /// * `shape` - The shape (already transformed) to be checked for collisions
242    /// * `filter` - Hazard filter to be applied
243    pub fn poly_collides(&self, shape: &SPolygon, filter: &impl HazardFilter) -> bool {
244        match self.bbox.relation_to(shape.bbox) {
245            //Not fully inside bbox => definite collision
246            GeoRelation::Disjoint | GeoRelation::Enclosed | GeoRelation::Intersecting => true,
247            GeoRelation::Surrounding => {
248                self.poly_collides_by_edge_intersection(shape, filter)
249                    || self.poly_collides_by_containment(shape, filter)
250            }
251        }
252    }
253
254    /// Checks whether a surrogate collides with any of the (relevant) hazards.
255    /// # Arguments
256    /// * `base_surrogate` - The (untransformed) surrogate to be checked for collisions
257    /// * `transform` - The transformation to be applied to the surrogate
258    /// * `filter` - Hazard filter to be applied
259    pub fn surrogate_collides(
260        &self,
261        base_surrogate: &SPSurrogate,
262        transform: &Transformation,
263        filter: &impl HazardFilter,
264    ) -> bool {
265        for pole in base_surrogate.ff_poles() {
266            let t_pole = pole.transform_clone(transform);
267            if self.quadtree.collides(&t_pole, filter).is_some() {
268                return true;
269            }
270        }
271        for pier in base_surrogate.ff_piers() {
272            let t_pier = pier.transform_clone(transform);
273            if self.quadtree.collides(&t_pier, filter).is_some() {
274                return true;
275            }
276        }
277        false
278    }

280    /// Checks whether a point definitely collides with any of the (relevant) hazards.
281    /// Only fully hazardous nodes in the quadtree are considered.
282    pub fn point_definitely_collides_with(&self, point: &Point, entity: HazardEntity) -> Tribool {
283        match self.bbox.collides_with(point) {
284            false => Tribool::Indeterminate, //point is outside the quadtree, so no information available
285            true => self.quadtree.definitely_collides_with(point, entity),
286        }
287    }

289    /// Checks whether an edge definitely collides with any of the (relevant) hazards.
290    /// Only fully hazardous nodes in the quadtree are considered.
291    pub fn edge_definitely_collides(&self, edge: &Edge, filter: &impl HazardFilter) -> Tribool {
292        match !self.bbox.collides_with(&edge.start) || !self.bbox.collides_with(&edge.end) {
293            true => Tribool::True, //if either the start or end of the edge is outside the quadtree, it definitely collides
294            false => self.quadtree.definitely_collides(edge, filter),
295        }
296    }

298    /// Checks whether a circle definitely collides with any of the (relevant) hazards.
299    /// Only fully hazardous nodes in the quadtree are considered.
300    pub fn circle_definitely_collides(
301        &self,
302        circle: &Circle,
303        filter: &impl HazardFilter,
304    ) -> Tribool {
305        match self.bbox.collides_with(&circle.center) {
306            false => Tribool::True, //outside the quadtree, so definitely collides
307            true => self.quadtree.definitely_collides(circle, filter),

321    fn poly_collides_by_containment(&self, shape: &SPolygon, filter: &impl HazardFilter) -> bool {
322        //collect all active and non-ignored hazards
323        self.all_hazards()
324            .filter(|h| h.active && !filter.is_irrelevant(&h.entity))
325            .any(|haz| self.poly_or_hazard_are_contained(shape, haz))
326    }

328    pub fn poly_or_hazard_are_contained(&self, shape: &SPolygon, haz: &Hazard) -> bool {
329        //Due to possible fp issues, we check if the bboxes are "almost" related
330        //"almost" meaning that, when edges are very close together, they are considered equal.
331        //Some relations which would normally be seen as Intersecting are now being considered Enclosed/Surrounding
332        let haz_shape = haz.shape.as_ref();
333        let bbox_relation = haz_shape.bbox.almost_relation_to(shape.bbox);
334
335        let (s_mu, s_omega) = match bbox_relation {
336            GeoRelation::Surrounding => (shape, haz_shape), //inclusion possible
337            GeoRelation::Enclosed => (haz_shape, shape),    //inclusion possible
338            GeoRelation::Disjoint | GeoRelation::Intersecting => {
339                //no inclusion is possible
340                return match haz.entity.position() {
341                    GeoPosition::Interior => false,
342                    GeoPosition::Exterior => true,
343                };
344            }
345        };
346
347        if std::ptr::eq(haz_shape, s_omega) {
348            //s_omega is registered in the quadtree.
349            //maybe the quadtree can help us.
350            if let Ok(collides) = self
351                .quadtree
352                .definitely_collides_with(&s_mu.poi.center, haz.entity)
353                .try_into()
354            {
355                return collides;
356            }
357        }
358        let inclusion = s_omega.collides_with(&s_mu.poi.center);
359
360        match haz.entity.position() {
361            GeoPosition::Interior => inclusion,
362            GeoPosition::Exterior => !inclusion,
363        }
364    }

366    /// Collects all hazards with which the polygon collides and reports them to the detector.
367    pub fn collect_poly_collisions(&self, shape: &SPolygon, detector: &mut impl HazardDetector) {
368        if self.bbox.relation_to(shape.bbox) != GeoRelation::Surrounding {
369            detector.push(HazardEntity::Exterior)
370        }
371
372        //collect all colliding entities due to edge intersection
373        shape
374            .edge_iter()
375            .for_each(|e| self.quadtree.collect_collisions(&e, detector));
376
377        //collect all colliding entities due to containment
378        self.all_hazards().filter(|h| h.active).for_each(|h| {
379            if !detector.contains(&h.entity) && self.poly_or_hazard_are_contained(shape, h) {
380                detector.push(h.entity);
381            }
382        });
383    }

385    /// Collects all hazards with which the surrogate collides and reports them to the detector.
386    pub fn collect_surrogate_collisions(
387        &self,
388        base_surrogate: &SPSurrogate,
389        transform: &Transformation,
390        detector: &mut impl HazardDetector,
391    ) {
392        for pole in base_surrogate.ff_poles() {
393            let t_pole = pole.transform_clone(transform);
394            self.quadtree.collect_collisions(&t_pole, detector)
395        }
396        for pier in base_surrogate.ff_piers() {
397            let t_pier = pier.transform_clone(transform);
398            self.quadtree.collect_collisions(&t_pier, detector);
399        }
400    }
