17/// Converts external representations of items and containers into internal ones.
18#[derive(Clone, Debug, Copy)]
19pub struct Importer {
20    pub shape_modify_config: ShapeModifyConfig,
21    pub cde_config: CDEConfig,
22}

24impl Importer {
25    /// Creates a new instance with the given configuration.
26    ///
27    /// * `cde_config` - Configuration for the CDE (Collision Detection Engine).
28    /// * `poly_simpl_tolerance` - See [`ShapeModifyConfig::simplify_tolerance`].
29    /// * `min_item_separation` - Optional minimum separation distance between items and any other hazard. If enabled, every hazard is inflated/deflated by half this value. See [`ShapeModifyConfig::offset`].
30    pub fn new(
31        cde_config: CDEConfig,
32        poly_simpl_tolerance: Option<f32>,
33        min_item_separation: Option<f32>,
34    ) -> Importer {
35        Importer {
36            shape_modify_config: ShapeModifyConfig {
37                offset: min_item_separation.map(|f| f / 2.0),
38                simplify_tolerance: poly_simpl_tolerance,
39            },
40            cde_config,
41        }
42    }

44    pub fn import_item(&self, ext_item: &ExtItem) -> Result<Item> {
45        let original_shape = {
46            let shape = match &ext_item.shape {
47                ExtShape::Rectangle {
48                    x_min,
49                    y_min,
50                    width,
51                    height,
52                } => {
53                    let rect = Rect::try_new(*x_min, *y_min, x_min + width, y_min + height)?;
54                    SPolygon::from(rect)
55                }
56                ExtShape::SimplePolygon(esp) => import_simple_polygon(esp)?,
57                ExtShape::Polygon(ep) => {
58                    warn!("No native support for polygons yet, ignoring the holes");
59                    import_simple_polygon(&ep.outer)?
60                }
61                ExtShape::MultiPolygon(_) => {
62                    bail!("No support for multipolygons yet")
63                }
64            };
65            OriginalShape {
66                pre_transform: centering_transformation(&shape),
67                shape,
68                modify_mode: ShapeModifyMode::Inflate,
69                modify_config: self.shape_modify_config,
70            }
71        };
72
73        let base_quality = ext_item.min_quality;
74
75        let allowed_orientations = match ext_item.allowed_orientations.as_ref() {
76            Some(a_o) => {
77                if a_o.is_empty() || (a_o.len() == 1 && a_o[0] == 0.0) {
78                    RotationRange::None
79                } else {
80                    RotationRange::Discrete(a_o.iter().map(|angle| angle.to_radians()).collect())
81                }
82            }
83            None => RotationRange::Continuous,
84        };
85
86        Item::new(
87            ext_item.id as usize,
88            original_shape,
89            allowed_orientations,
90            base_quality,
91            self.cde_config.item_surrogate_config,
92        )
93    }

95    pub fn import_container(&self, ext_cont: &ExtContainer) -> Result<Container> {
96        assert!(
97            ext_cont.zones.iter().all(|zone| zone.quality < N_QUALITIES),
98            "All quality zones must have lower quality than N_QUALITIES, set N_QUALITIES to a higher value if required"
99        );
100
101        let original_outer = {
102            let outer = match &ext_cont.shape {
103                ExtShape::Rectangle {
104                    x_min,
105                    y_min,
106                    width,
107                    height,
108                } => Rect::try_new(*x_min, *y_min, x_min + width, y_min + height)?.into(),
109                ExtShape::SimplePolygon(esp) => import_simple_polygon(esp)?,
110                ExtShape::Polygon(ep) => import_simple_polygon(&ep.outer)?,
111                ExtShape::MultiPolygon(_) => {
112                    bail!("No support for multipolygon shapes yet")
113                }
114            };
115            OriginalShape {
116                shape: outer,
117                pre_transform: DTransformation::empty(),
118                modify_mode: ShapeModifyMode::Deflate,
119                modify_config: self.shape_modify_config,
120            }
121        };
122
123        let holes = match &ext_cont.shape {
124            ExtShape::SimplePolygon(_) | ExtShape::Rectangle { .. } => vec![],
125            ExtShape::Polygon(jp) => {
126                let json_holes = &jp.inner;
127                json_holes
128                    .iter()
129                    .map(import_simple_polygon)
130                    .collect::<Result<Vec<SPolygon>>>()?
131            }
132            ExtShape::MultiPolygon(_) => {
133                unimplemented!("No support for multipolygon shapes yet")
134            }
135        };
136
137        let mut shapes_inferior_qzones = (0..N_QUALITIES)
138            .map(|q| {
139                ext_cont
140                    .zones
141                    .iter()
142                    .filter(|zone| zone.quality == q)
143                    .map(|zone| match &zone.shape {
144                        ExtShape::Rectangle {
145                            x_min,
146                            y_min,
147                            width,
148                            height,
149                        } => Rect::try_new(*x_min, *y_min, x_min + width, y_min + height)
150                            .map(|r| r.into()),
151                        ExtShape::SimplePolygon(esp) => import_simple_polygon(esp),
152                        ExtShape::Polygon(_) => {
153                            unimplemented!("No support for polygon to simplepolygon conversion yet")
154                        }
155                        ExtShape::MultiPolygon(_) => {
156                            unimplemented!("No support for multipolygon shapes yet")
157                        }
158                    })
159                    .collect::<Result<Vec<SPolygon>>>()
160            })
161            .collect::<Result<Vec<Vec<SPolygon>>>>()?;
162
163        //merge the container holes with quality == 0
164        shapes_inferior_qzones[0].extend(holes);
165
166        //convert the shapes to inferior quality zones
167        let quality_zones = shapes_inferior_qzones
168            .into_iter()
169            .enumerate()
170            .map(|(q, zone_shapes)| {
171                let original_shapes = zone_shapes
172                    .into_iter()
173                    .map(|s| OriginalShape {
174                        shape: s,
175                        pre_transform: DTransformation::empty(),
176                        modify_mode: ShapeModifyMode::Inflate,
177                        modify_config: self.shape_modify_config,
178                    })
179                    .collect_vec();
180                InferiorQualityZone::new(q, original_shapes)
181            })
182            .collect::<Result<Vec<InferiorQualityZone>>>()?;
183
184        Container::new(
185            ext_cont.id as usize,
186            original_outer,
187            quality_zones,
188            self.cde_config,
189        )
190    }
191}

193pub fn import_simple_polygon(sp: &ExtSPolygon) -> Result<SPolygon> {
194    let mut points = sp.0.iter().map(|(x, y)| Point(*x, *y)).collect_vec();
195    //Strip the last vertex if it is the same as the first one
196    if points.len() > 1 && points[0] == points[points.len() - 1] {
197        points.pop();
198    }
199    //Remove duplicates that are consecutive (e.g. [1, 2, 2, 3] -> [1, 2, 3])
200    eliminate_degenerate_points(&mut points);
201    //Bail if there are any non-consecutive duplicates.
202    if points.len() != points.iter().unique().count() {
203        bail!("Simple polygon has non-consecutive duplicate vertices");
204    }
205    SPolygon::new(points)
206}

208/// Returns a transformation that translates the shape's centroid to the origin.
209pub fn centering_transformation(shape: &SPolygon) -> DTransformation {
210    let Point(cx, cy) = shape.centroid();
211    DTransformation::new(0.0, (-cx, -cy))
212}

214/// Converts an external transformation (applicable to the original shapes) to an internal transformation (used within `jagua-rs`).
215///
216/// * `ext_transf` - The external transformation.
217/// * `pre_transf` - The transformation that was applied to the original shape to derive the internal representation.
218pub fn ext_to_int_transformation(
219    ext_transf: &DTransformation,
220    pre_transf: &DTransformation,
221) -> DTransformation {
222    //1. undo pre-transform
223    //2. do the absolute transformation
224
225    Transformation::empty()
226        .transform(&pre_transf.compose().inverse())
227        .transform_from_decomposed(ext_transf)
228        .decompose()
229}

231pub fn eliminate_degenerate_points(points: &mut Vec<Point>) {
232    let mut indices_to_remove = vec![];
233    let n_points = points.len();
234    for i in 0..n_points {
235        let j = (i + 1) % n_points;
236        let p_i = points[i];
237        let p_j = points[j];
238        if approx_eq!(f32, p_i.0, p_j.0) && approx_eq!(f32, p_i.1, p_j.1) {
239            //points are equal, mark for removal
240            indices_to_remove.push(i);
241        }
242    }
243    //remove points in reverse order to avoid shifting indices
244    indices_to_remove.sort_unstable_by(|a, b| b.cmp(a));
245    for index in indices_to_remove {
246        if index < points.len() {
247            let j = (index + 1) % points.len();
248            warn!(
249                "[IMPORT] degenerate point of input simple polygon eliminated (idx: {}, {:?}, {:?})",
250                index, points[index], points[j]
251            );
252            points.remove(index);
253        }
254    }
