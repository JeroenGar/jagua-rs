66impl core::fmt::Display for Error {
67    fn fmt(&self, f: &mut core::fmt::Formatter) -> core::fmt::Result {
68        use self::Error::*;
69
70        match *self {
71            CommentClosingParenWithoutOpen => f.write_str(
72                "found closing parenthesis in comment with \
73                 no matching opening parenthesis",
74            ),
75            CommentOpeningParenWithoutClose => f.write_str(
76                "found opening parenthesis in comment with \
77                 no matching closing parenthesis",
78            ),
79            CommentTooManyNestedParens => {
80                f.write_str("found too many nested parenthesis in comment")
81            }
82            Empty => {
83                f.write_str("expected RFC 2822 datetime, but got empty string")
84            }
85            EmptyAfterWhitespace => f.write_str(
86                "expected RFC 2822 datetime, but got empty string \
87                 after trimming leading whitespace",
88            ),
89            EndOfInputComma => f.write_str(
90                "expected comma after parsed weekday in \
91                 RFC 2822 datetime, but found end of input instead",
92            ),
93            EndOfInputDay => {
94                f.write_str("expected numeric day, but found end of input")
95            }
96            EndOfInputHour => {
97                f.write_str("expected two digit hour, but found end of input")
98            }
99            EndOfInputMinute => f.write_str(
100                "expected two digit minute, but found end of input",
101            ),
102            EndOfInputMonth => f.write_str(
103                "expected abbreviated month name, but found end of input",
104            ),
105            EndOfInputOffset => f.write_str(
106                "expected sign for time zone offset, \
107                 (or a legacy time zone name abbreviation), \
108                 but found end of input",
109            ),
110            EndOfInputSecond => f.write_str(
111                "expected two digit second, but found end of input",
112            ),
113            EndOfInputTimeSeparator => f.write_str(
114                "expected time separator of `:`, but found end of input",
115            ),
116            FailedTimestamp => f.write_str(
117                "failed to parse RFC 2822 datetime into Jiff timestamp",
118            ),
119            FailedZoned => f.write_str(
120                "failed to parse RFC 2822 datetime into Jiff zoned datetime",
121            ),
122            InconsistentWeekday { parsed, from_date } => write!(
123                f,
124                "found parsed weekday of `{parsed:?}`, \
125                 but parsed datetime has weekday `{from_date:?}`",
126            ),
127            InvalidDate => f.write_str("invalid date"),
128            InvalidHour => f.write_str("invalid hour"),
129            InvalidMinute => f.write_str("invalid minute"),
130            InvalidMonth => f.write_str(
131                "expected abbreviated month name, \
132                 but did not recognize a valid abbreviated month name",
133            ),
134            InvalidObsoleteOffset => f.write_str(
135                "expected obsolete RFC 2822 time zone abbreviation, \
136                 but did not recognize a valid abbreviation",
137            ),
138            InvalidOffsetHour => f.write_str("invalid time zone offset hour"),
139            InvalidOffsetMinute => {
140                f.write_str("invalid time zone offset minute")
141            }
142            InvalidSecond => f.write_str("invalid second"),
143            InvalidWeekday { got_non_digit } => write!(
144                f,
145                "expected day at beginning of RFC 2822 datetime \
146                 since first non-whitespace byte, `{first}`, \
147                 is not a digit, but did not recognize a valid \
148                 weekday abbreviation",
149                first = escape::Byte(got_non_digit),
150            ),
151            InvalidYear => f.write_str("invalid year"),
152            NegativeYear => f.write_str(
153                "datetime has negative year, \
154                 which cannot be formatted with RFC 2822",
155            ),
156            ParseDay => f.write_str("failed to parse day"),
157            ParseHour => f.write_str(
158                "failed to parse hour (expects a two digit integer)",
159            ),
160            ParseMinute => f.write_str(
161                "failed to parse minute (expects a two digit integer)",
162            ),
163            ParseOffsetHour => {
164                f.write_str("failed to parse hours from time zone offset")
165            }
166            ParseOffsetMinute => {
167                f.write_str("failed to parse minutes from time zone offset")
168            }
169            ParseSecond => f.write_str(
170                "failed to parse second (expects a two digit integer)",
171            ),
172            ParseYear => f.write_str(
173                "failed to parse year \
174                 (expects a two, three or four digit integer)",
175            ),
176            TooShortMonth { len } => write!(
177                f,
178                "expected abbreviated month name, but remaining input \
179                 is too short (remaining bytes is {len})",
180            ),
181            TooShortOffset => write!(
182                f,
183                "expected at least four digits for time zone offset \
184                 after sign, but found fewer than four bytes remaining",
185            ),
186            TooShortWeekday { got_non_digit, len } => write!(
187                f,
188                "expected day at beginning of RFC 2822 datetime \
189                 since first non-whitespace byte, `{first}`, \
190                 is not a digit, but given string is too short \
191                 (length is {len})",
192                first = escape::Byte(got_non_digit),
193            ),
194            TooShortYear { len } => write!(
195                f,
196                "expected at least two ASCII digits for parsing \
197                 a year, but only found {len}",
198            ),
199            UnexpectedByteComma { byte } => write!(
200                f,
201                "expected comma after parsed weekday in \
202                 RFC 2822 datetime, but found `{got}` instead",
203                got = escape::Byte(byte),
204            ),
205            UnexpectedByteTimeSeparator { byte } => write!(
206                f,
207                "expected time separator of `:`, but found `{got}`",
208                got = escape::Byte(byte),
209            ),
210            WhitespaceAfterDay => {
211                f.write_str("expected whitespace after parsing day")
212            }
213            WhitespaceAfterMonth => f.write_str(
214                "expected whitespace after parsing abbreviated month name",
215            ),
216            WhitespaceAfterTime => f.write_str(
217                "expected whitespace after parsing time: \
218                 expected at least one whitespace character \
219                 (space or tab), but found none",
220            ),
221            WhitespaceAfterTimeForObsoleteOffset => f.write_str(
222                "expected obsolete RFC 2822 time zone abbreviation, \
223                 but found no remaining non-whitespace characters \
224                 after time",
225            ),
226            WhitespaceAfterYear => {
227                f.write_str("expected whitespace after parsing year")
228            }
229        }
230    }
