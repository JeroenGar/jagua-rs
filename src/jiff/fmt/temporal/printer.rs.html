52    pub(super) fn print_zoned<W: Write>(
53        &self,
54        zdt: &Zoned,
55        mut wtr: W,
56    ) -> Result<(), Error> {
57        let timestamp = zdt.timestamp();
58        let tz = zdt.time_zone();
59        let offset = tz.to_offset(timestamp);
60        let dt = offset.to_datetime(timestamp);
61        self.print_datetime(&dt, &mut wtr)?;
62        if tz.is_unknown() {
63            wtr.write_str("Z[Etc/Unknown]")?;
64        } else {
65            self.print_offset_rounded(&offset, &mut wtr)?;
66            self.print_time_zone_annotation(&tz, &offset, &mut wtr)?;
67        }
68        Ok(())
69    }

72        &self,
73        timestamp: &Timestamp,
74        offset: Option<Offset>,
75        mut wtr: W,
76    ) -> Result<(), Error> {
77        let Some(offset) = offset else {
78            let dt = TimeZone::UTC.to_datetime(*timestamp);
79            self.print_datetime(&dt, &mut wtr)?;
80            self.print_zulu(&mut wtr)?;
81            return Ok(());

84        self.print_datetime(&dt, &mut wtr)?;
85        self.print_offset_rounded(&offset, &mut wtr)?;

90    pub(super) fn print_datetime<W: Write>(
91        &self,
92        dt: &DateTime,
93        mut wtr: W,
94    ) -> Result<(), Error> {
95        self.print_date(&dt.date(), &mut wtr)?;

105    /// Formats the given date into the writer given.
106    pub(super) fn print_date<W: Write>(
107        &self,
108        date: &Date,
109        mut wtr: W,
110    ) -> Result<(), Error> {
111        static FMT_YEAR_POSITIVE: IntegerFormatter =
112            IntegerFormatter::new().padding(4);
113        static FMT_YEAR_NEGATIVE: IntegerFormatter =
114            IntegerFormatter::new().padding(6);
115        static FMT_TWO: IntegerFormatter = IntegerFormatter::new().padding(2);
116
117        if date.year() >= 0 {
118            wtr.write_int(&FMT_YEAR_POSITIVE, date.year())?;
119        } else {
120            wtr.write_int(&FMT_YEAR_NEGATIVE, date.year())?;
121        }
122        wtr.write_str("-")?;
123        wtr.write_int(&FMT_TWO, date.month())?;
124        wtr.write_str("-")?;
125        wtr.write_int(&FMT_TWO, date.day())?;
126        Ok(())
127    }

129    /// Formats the given time into the writer given.
130    pub(super) fn print_time<W: Write>(
131        &self,
132        time: &Time,
133        mut wtr: W,
134    ) -> Result<(), Error> {
135        static FMT_TWO: IntegerFormatter = IntegerFormatter::new().padding(2);
136        static FMT_FRACTION: FractionalFormatter = FractionalFormatter::new();
137
138        wtr.write_int(&FMT_TWO, time.hour())?;
139        wtr.write_str(":")?;
140        wtr.write_int(&FMT_TWO, time.minute())?;
141        wtr.write_str(":")?;
142        wtr.write_int(&FMT_TWO, time.second())?;
143        let fractional_nanosecond = time.subsec_nanosecond();
144        if self.precision.map_or(fractional_nanosecond != 0, |p| p > 0) {
145            wtr.write_str(".")?;
146            wtr.write_fraction(
147                &FMT_FRACTION.precision(self.precision),
148                fractional_nanosecond.unsigned_abs(),
149            )?;
150        }
151        Ok(())
152    }

154    /// Formats the given time zone into the writer given.
155    pub(super) fn print_time_zone<W: Write>(
156        &self,
157        tz: &TimeZone,
158        mut wtr: W,
159    ) -> Result<(), Error> {
160        if let Some(iana_name) = tz.iana_name() {
161            return wtr.write_str(iana_name);
162        }
163        if tz.is_unknown() {
164            return wtr.write_str("Etc/Unknown");
165        }
166        if let Ok(offset) = tz.to_fixed_offset() {
167            return self.print_offset_full_precision(&offset, wtr);
168        }
169        // We get this on `alloc` because we format the POSIX time zone into a
170        // `String` first. See the note below.
171        //
172        // This is generally okay because there is no current (2025-02-28) way
173        // to create a `TimeZone` that is *only* a POSIX time zone in core-only
174        // environments. (All you can do is create a TZif time zone, which may
175        // contain a POSIX time zone, but `tz.posix_tz()` would still return
176        // `None` in that case.)
177        #[cfg(feature = "alloc")]
178        {
179            if let Some(posix_tz) = tz.posix_tz() {
180                // This is pretty unfortunate, but at time of writing, I
181                // didn't see an easy way to make the `Display` impl for
182                // `PosixTimeZone` automatically work with
183                // `jiff::fmt::Write` without allocating a new string. As
184                // far as I can see, I either have to duplicate the code or
185                // make it generic in some way. I judged neither to be worth
186                // doing for such a rare case. ---AG
187                let s = alloc::string::ToString::to_string(posix_tz);
188                return wtr.write_str(&s);
189            }
190        }
191        // Ideally this never actually happens, but it can, and there
192        // are likely system configurations out there in which it does.
193        // I can imagine "lightweight" installations that just have a
194        // `/etc/localtime` as a TZif file that doesn't point to any IANA time
195        // zone. In which case, serializing a time zone probably doesn't make
196        // much sense.
197        //
198        // Anyway, if you're seeing this error and think there should be a
199        // different behavior, please file an issue.
200        Err(Error::from(E::PrintTimeZoneFailure))
201    }

203    pub(super) fn print_pieces<W: Write>(
204        &self,
205        pieces: &Pieces,
206        mut wtr: W,
207    ) -> Result<(), Error> {
208        if let Some(time) = pieces.time() {
209            let dt = DateTime::from_parts(pieces.date(), time);
210            self.print_datetime(&dt, &mut wtr)?;
211            if let Some(poffset) = pieces.offset() {
212                self.print_pieces_offset(&poffset, &mut wtr)?;
213            }
214        } else if let Some(poffset) = pieces.offset() {
215            // In this case, we have an offset but no time component. Since
216            // `2025-01-02-05:00` isn't valid, we forcefully write out the
217            // default time (which is what would be assumed anyway).
218            let dt = DateTime::from_parts(pieces.date(), Time::midnight());
219            self.print_datetime(&dt, &mut wtr)?;
220            self.print_pieces_offset(&poffset, &mut wtr)?;
221        } else {
222            // We have no time and no offset, so we can just write the date.
223            // It's okay to write this followed by an annotation, e.g.,
224            // `2025-01-02[America/New_York]` or even `2025-01-02[-05:00]`.
225            self.print_date(&pieces.date(), &mut wtr)?;
226        }
227        // For the time zone annotation, a `Pieces` gives us the annotation
228        // name or offset directly, where as with `Zoned`, we have a
229        // `TimeZone`. So we hand-roll our own formatter directly from the
230        // annotation.
231        if let Some(ann) = pieces.time_zone_annotation() {
232            // Note that we explicitly ignore `self.rfc9557` here, since with
233            // `Pieces`, the annotation has been explicitly provided. Also,
234            // at time of writing, `self.rfc9557` is always enabled anyway.
235            wtr.write_str("[")?;
236            if ann.is_critical() {
237                wtr.write_str("!")?;
238            }
239            match *ann.kind() {
240                TimeZoneAnnotationKind::Named(ref name) => {
241                    wtr.write_str(name.as_str())?
242                }
243                TimeZoneAnnotationKind::Offset(offset) => {
244                    self.print_offset_rounded(&offset, &mut wtr)?
245                }
246            }
247            wtr.write_str("]")?;
248        }
249        Ok(())
250    }

252    pub(super) fn print_iso_week_date<W: Write>(
253        &self,
254        iso_week_date: &ISOWeekDate,
255        mut wtr: W,
256    ) -> Result<(), Error> {
257        static FMT_YEAR_POSITIVE: IntegerFormatter =
258            IntegerFormatter::new().padding(4);
259        static FMT_YEAR_NEGATIVE: IntegerFormatter =
260            IntegerFormatter::new().padding(6);
261        static FMT_TWO: IntegerFormatter = IntegerFormatter::new().padding(2);
262        static FMT_ONE: IntegerFormatter = IntegerFormatter::new().padding(1);
263
264        if iso_week_date.year() >= 0 {
265            wtr.write_int(&FMT_YEAR_POSITIVE, iso_week_date.year())?;
266        } else {
267            wtr.write_int(&FMT_YEAR_NEGATIVE, iso_week_date.year())?;
268        }
269        wtr.write_str("-")?;
270        wtr.write_char(if self.lowercase { 'w' } else { 'W' })?;
271        wtr.write_int(&FMT_TWO, iso_week_date.week())?;
272        wtr.write_str("-")?;
273        wtr.write_int(
274            &FMT_ONE,
275            iso_week_date.weekday().to_monday_one_offset(),
276        )?;
277        Ok(())
278    }

280    /// Formats the given "pieces" offset into the writer given.
281    fn print_pieces_offset<W: Write>(
282        &self,
283        poffset: &PiecesOffset,
284        mut wtr: W,
285    ) -> Result<(), Error> {
286        match *poffset {
287            PiecesOffset::Zulu => self.print_zulu(wtr),
288            PiecesOffset::Numeric(ref noffset) => {
289                if noffset.offset().is_zero() && noffset.is_negative() {
290                    wtr.write_str("-00:00")
291                } else {
292                    self.print_offset_rounded(&noffset.offset(), wtr)
293                }
294            }
295        }
296    }

298    /// Formats the given offset into the writer given.
299    ///
300    /// If the given offset has non-zero seconds, then they are rounded to
301    /// the nearest minute.
302    fn print_offset_rounded<W: Write>(
303        &self,
304        offset: &Offset,
305        mut wtr: W,
306    ) -> Result<(), Error> {
307        static FMT_TWO: IntegerFormatter = IntegerFormatter::new().padding(2);
308
309        wtr.write_str(if offset.is_negative() { "-" } else { "+" })?;
310        let mut hours = offset.part_hours_ranged().abs().get();
311        let mut minutes = offset.part_minutes_ranged().abs().get();
312        // RFC 3339 requires that time zone offsets are an integral number
313        // of minutes. While rounding based on seconds doesn't seem clearly
314        // indicated, the `1937-01-01T12:00:27.87+00:20` example seems
315        // to suggest that the number of minutes should be "as close as
316        // possible" to the actual offset. So we just do basic rounding
317        // here.
318        if offset.part_seconds_ranged().abs() >= C(30) {
319            if minutes == 59 {
320                hours = hours.saturating_add(1);
321                minutes = 0;
322            } else {
323                minutes = minutes.saturating_add(1);
324            }
325        }
326        wtr.write_int(&FMT_TWO, hours)?;
327        wtr.write_str(":")?;
328        wtr.write_int(&FMT_TWO, minutes)?;
329        Ok(())
330    }

332    /// Formats the given offset into the writer given.
333    ///
334    /// If the given offset has non-zero seconds, then they are emitted as a
335    /// third `:`-delimited component of the offset. If seconds are zero, then
336    /// only the hours and minute components are emitted.
337    fn print_offset_full_precision<W: Write>(
338        &self,
339        offset: &Offset,
340        mut wtr: W,
341    ) -> Result<(), Error> {
342        static FMT_TWO: IntegerFormatter = IntegerFormatter::new().padding(2);
343
344        wtr.write_str(if offset.is_negative() { "-" } else { "+" })?;
345        let hours = offset.part_hours_ranged().abs().get();
346        let minutes = offset.part_minutes_ranged().abs().get();
347        let seconds = offset.part_seconds_ranged().abs().get();
348        wtr.write_int(&FMT_TWO, hours)?;
349        wtr.write_str(":")?;
350        wtr.write_int(&FMT_TWO, minutes)?;
351        if seconds > 0 {
352            wtr.write_str(":")?;
353            wtr.write_int(&FMT_TWO, seconds)?;
354        }
355        Ok(())
356    }

358    /// Prints the "zulu" indicator.
359    ///
360    /// This should only be used when the offset is not known. For example,
361    /// when printing a `Timestamp`.
362    fn print_zulu<W: Write>(&self, mut wtr: W) -> Result<(), Error> {
363        wtr.write_str(if self.lowercase { "z" } else { "Z" })
364    }

366    /// Formats the given time zone name into the writer given as an RFC 9557
367    /// time zone annotation.
368    ///
369    /// This is a no-op when RFC 9557 support isn't enabled. And when the given
370    /// time zone is not an IANA time zone name, then the offset is printed
371    /// instead. (This means the offset will be printed twice, which is indeed
372    /// an intended behavior of RFC 9557 for cases where a time zone name is
373    /// not used or unavailable.)
374    fn print_time_zone_annotation<W: Write>(
375        &self,
376        time_zone: &TimeZone,
377        offset: &Offset,
378        mut wtr: W,
379    ) -> Result<(), Error> {
380        if !self.rfc9557 {
381            return Ok(());
382        }
383        wtr.write_str("[")?;
384        if let Some(iana_name) = time_zone.iana_name() {
385            wtr.write_str(iana_name)?;
386        } else {
387            self.print_offset_rounded(offset, &mut wtr)?;
388        }
389        wtr.write_str("]")?;
390        Ok(())
391    }

422    /// Print the given span to the writer given.
423    ///
424    /// This only returns an error when the given writer returns an error.
425    pub(super) fn print_span<W: Write>(
426        &self,
427        span: &Span,
428        mut wtr: W,
429    ) -> Result<(), Error> {
430        static FMT_INT: IntegerFormatter = IntegerFormatter::new();
431        static FMT_FRACTION: FractionalFormatter = FractionalFormatter::new();
432
433        if span.is_negative() {
434            wtr.write_str("-")?;
435        }
436        wtr.write_str("P")?;
437
438        let mut non_zero_greater_than_second = false;
439        if span.get_years_ranged() != C(0) {
440            wtr.write_int(&FMT_INT, span.get_years_ranged().get().abs())?;
441            wtr.write_char(self.label('Y'))?;
442            non_zero_greater_than_second = true;
443        }
444        if span.get_months_ranged() != C(0) {
445            wtr.write_int(&FMT_INT, span.get_months_ranged().get().abs())?;
446            wtr.write_char(self.label('M'))?;
447            non_zero_greater_than_second = true;
448        }
449        if span.get_weeks_ranged() != C(0) {
450            wtr.write_int(&FMT_INT, span.get_weeks_ranged().get().abs())?;
451            wtr.write_char(self.label('W'))?;
452            non_zero_greater_than_second = true;
453        }
454        if span.get_days_ranged() != C(0) {
455            wtr.write_int(&FMT_INT, span.get_days_ranged().get().abs())?;
456            wtr.write_char(self.label('D'))?;
457            non_zero_greater_than_second = true;
458        }
459
460        let mut printed_time_prefix = false;
461        if span.get_hours_ranged() != C(0) {
462            if !printed_time_prefix {
463                wtr.write_str("T")?;
464                printed_time_prefix = true;
465            }
466            wtr.write_int(&FMT_INT, span.get_hours_ranged().get().abs())?;
467            wtr.write_char(self.label('H'))?;
468            non_zero_greater_than_second = true;
469        }
470        if span.get_minutes_ranged() != C(0) {
471            if !printed_time_prefix {
472                wtr.write_str("T")?;
473                printed_time_prefix = true;
474            }
475            wtr.write_int(&FMT_INT, span.get_minutes_ranged().get().abs())?;
476            wtr.write_char(self.label('M'))?;
477            non_zero_greater_than_second = true;
478        }
479
480        // ISO 8601 (and Temporal) don't support writing out milliseconds,
481        // microseconds or nanoseconds as separate components like for all
482        // the other units. Instead, they must be incorporated as fractional
483        // seconds. But we only want to do that work if we need to.
484        let (seconds, millis, micros, nanos) = (
485            span.get_seconds_ranged().abs(),
486            span.get_milliseconds_ranged().abs(),
487            span.get_microseconds_ranged().abs(),
488            span.get_nanoseconds_ranged().abs(),
489        );
490        if (seconds != C(0) || !non_zero_greater_than_second)
491            && millis == C(0)
492            && micros == C(0)
493            && nanos == C(0)
494        {
495            if !printed_time_prefix {
496                wtr.write_str("T")?;
497            }
498            wtr.write_int(&FMT_INT, seconds.get())?;
499            wtr.write_char(self.label('S'))?;
500        } else if millis != C(0) || micros != C(0) || nanos != C(0) {
501            if !printed_time_prefix {
502                wtr.write_str("T")?;
503            }
504            // We want to combine our seconds, milliseconds, microseconds and
505            // nanoseconds into one single value in terms of nanoseconds. Then
506            // we can "balance" that out so that we have a number of seconds
507            // and a number of nanoseconds not greater than 1 second. (Which is
508            // our fraction.)
509            let combined_as_nanos =
510                t::SpanSecondsOrLowerNanoseconds::rfrom(nanos)
511                    + (t::SpanSecondsOrLowerNanoseconds::rfrom(micros)
512                        * t::NANOS_PER_MICRO)
513                    + (t::SpanSecondsOrLowerNanoseconds::rfrom(millis)
514                        * t::NANOS_PER_MILLI)
515                    + (t::SpanSecondsOrLowerNanoseconds::rfrom(seconds)
516                        * t::NANOS_PER_SECOND);
517            let fraction_second = t::SpanSecondsOrLower::rfrom(
518                combined_as_nanos / t::NANOS_PER_SECOND,
519            );
520            let fraction_nano = t::SubsecNanosecond::rfrom(
521                combined_as_nanos % t::NANOS_PER_SECOND,
522            );
523            wtr.write_int(&FMT_INT, fraction_second.get())?;
524            if fraction_nano != C(0) {
525                wtr.write_str(".")?;
526                wtr.write_fraction(
527                    &FMT_FRACTION,
528                    i32::from(fraction_nano).unsigned_abs(),
529                )?;
530            }
531            wtr.write_char(self.label('S'))?;
532        }
533        Ok(())
534    }

536    /// Print the given signed duration to the writer given.
537    ///
538    /// This only returns an error when the given writer returns an error.
539    pub(super) fn print_signed_duration<W: Write>(
540        &self,
541        dur: &SignedDuration,
542        mut wtr: W,
543    ) -> Result<(), Error> {
544        static FMT_INT: IntegerFormatter = IntegerFormatter::new();
545        static FMT_FRACTION: FractionalFormatter = FractionalFormatter::new();
546
547        let mut non_zero_greater_than_second = false;
548        if dur.is_negative() {
549            wtr.write_str("-")?;
550        }
551        wtr.write_str("PT")?;
552
553        let mut secs = dur.as_secs();
554        // OK because subsec_nanos -999_999_999<=nanos<=999_999_999.
555        let nanos = dur.subsec_nanos().abs();
556        // OK because guaranteed to be bigger than i64::MIN.
557        let hours = (secs / (60 * 60)).abs();
558        secs %= 60 * 60;
559        // OK because guaranteed to be bigger than i64::MIN.
560        let minutes = (secs / 60).abs();
561        // OK because guaranteed to be bigger than i64::MIN.
562        secs = (secs % 60).abs();
563        if hours != 0 {
564            wtr.write_int(&FMT_INT, hours)?;
565            wtr.write_char(self.label('H'))?;
566            non_zero_greater_than_second = true;
567        }
568        if minutes != 0 {
569            wtr.write_int(&FMT_INT, minutes)?;
570            wtr.write_char(self.label('M'))?;
571            non_zero_greater_than_second = true;
572        }
573        if (secs != 0 || !non_zero_greater_than_second) && nanos == 0 {
574            wtr.write_int(&FMT_INT, secs)?;
575            wtr.write_char(self.label('S'))?;
576        } else if nanos != 0 {
577            wtr.write_int(&FMT_INT, secs)?;
578            wtr.write_str(".")?;
579            wtr.write_fraction(&FMT_FRACTION, nanos.unsigned_abs())?;
580            wtr.write_char(self.label('S'))?;
581        }
582        Ok(())
583    }

585    /// Print the given unsigned duration to the writer given.
586    ///
587    /// This only returns an error when the given writer returns an error.
588    pub(super) fn print_unsigned_duration<W: Write>(
589        &self,
590        dur: &core::time::Duration,
591        mut wtr: W,
592    ) -> Result<(), Error> {
593        static FMT_INT: IntegerFormatter = IntegerFormatter::new();
594        static FMT_FRACTION: FractionalFormatter = FractionalFormatter::new();
595
596        let mut non_zero_greater_than_second = false;
597        wtr.write_str("PT")?;
598
599        let mut secs = dur.as_secs();
600        let nanos = dur.subsec_nanos();
601        let hours = secs / (60 * 60);
602        secs %= 60 * 60;
603        let minutes = secs / 60;
604        secs = secs % 60;
605        if hours != 0 {
606            wtr.write_uint(&FMT_INT, hours)?;
607            wtr.write_char(self.label('H'))?;
608            non_zero_greater_than_second = true;
609        }
610        if minutes != 0 {
611            wtr.write_uint(&FMT_INT, minutes)?;
612            wtr.write_char(self.label('M'))?;
613            non_zero_greater_than_second = true;
614        }
615        if (secs != 0 || !non_zero_greater_than_second) && nanos == 0 {
616            wtr.write_uint(&FMT_INT, secs)?;
617            wtr.write_char(self.label('S'))?;
618        } else if nanos != 0 {
619            wtr.write_uint(&FMT_INT, secs)?;
620            wtr.write_str(".")?;
621            wtr.write_fraction(&FMT_FRACTION, nanos)?;
622            wtr.write_char(self.label('S'))?;
623        }
624        Ok(())
625    }
626
627    /// Converts the uppercase unit designator label to lowercase if this
628    /// printer is configured to use lowercase. Otherwise the label is returned
629    /// unchanged.
630    fn label(&self, upper: char) -> char {
631        debug_assert!(upper.is_ascii());
632        if self.lowercase {
633            upper.to_ascii_lowercase()
634        } else {
635            upper
636        }
637    }
