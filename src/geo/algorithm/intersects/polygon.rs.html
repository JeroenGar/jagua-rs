8impl<T> Intersects<Coord<T>> for Polygon<T>
9where
10    T: GeoNum,
11{
12    fn intersects(&self, p: &Coord<T>) -> bool {
13        coord_pos_relative_to_ring(*p, self.exterior()) != CoordPos::Outside
14            && self
15                .interiors()
16                .iter()
17                .all(|int| coord_pos_relative_to_ring(*p, int) != CoordPos::Inside)
18    }
19}

21symmetric_intersects_impl!(Polygon<T>, Point<T>);

23impl<T> Intersects<Line<T>> for Polygon<T>
24where
25    T: GeoNum,
26{
27    fn intersects(&self, line: &Line<T>) -> bool {
28        self.exterior().intersects(line)
29            || self.interiors().iter().any(|inner| inner.intersects(line))
30            || self.intersects(&line.start)
31            || self.intersects(&line.end)
32    }
33}
34symmetric_intersects_impl!(Line<T>, Polygon<T>);
35symmetric_intersects_impl!(Polygon<T>, LineString<T>);
36symmetric_intersects_impl!(Polygon<T>, MultiLineString<T>);

48impl<T> Intersects<Polygon<T>> for Polygon<T>
49where
50    T: GeoNum,
51{
52    fn intersects(&self, polygon: &Polygon<T>) -> bool {
53        if has_disjoint_bboxes(self, polygon) {
54            return false;
55        }
56
57        // self intersects (or contains) any line in polygon
58        self.intersects(polygon.exterior()) ||
59            polygon.interiors().iter().any(|inner_line_string| self.intersects(inner_line_string)) ||
60            // self is contained inside polygon
61            polygon.intersects(self.exterior())
62    }
63}

67impl<G, T> Intersects<G> for MultiPolygon<T>
68where
69    T: GeoNum,
70    Polygon<T>: Intersects<G>,
71    G: BoundingRect<T>,
72{
73    fn intersects(&self, rhs: &G) -> bool {
74        if has_disjoint_bboxes(self, rhs) {
75            return false;
76        }
77        self.iter().any(|p| p.intersects(rhs))
78    }
79}
