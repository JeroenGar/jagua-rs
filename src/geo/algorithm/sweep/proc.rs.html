14    pub(crate) fn new<I>(iter: I, is_simple: bool) -> Self
15    where
16        I: IntoIterator<Item = C>,
17    {
18        let iter = iter.into_iter();
19        let size = {
20            let (min_size, max_size) = iter.size_hint();
21            max_size.unwrap_or(min_size)
22        };
23
24        let mut sweep = Sweep {
25            events: BinaryHeap::with_capacity(size),
26            active_segments: Default::default(),
27            is_simple,
28        };
29        for cr in iter {
30            IMSegment::create_segment(cr, None, None, |ev| sweep.events.push(ev));
31        }
32
33        sweep
34    }

40    pub(super) fn next_event<F>(&mut self, mut cb: F) -> Option<SweepPoint<C::Scalar>>
41    where
42        F: for<'a> FnMut(&'a IMSegment<C>, EventType),
43    {
44        self.events.pop().map(|event| {
45            let pt = event.point;
46            self.handle_event(event, &mut cb);
47
48            pt
49        })
50    }

52    /// Process two adjacent segments.
53    ///
54    /// The first argument must be an active segment, and the other may or may not be.
55    /// Overlaps are chained from active -> other.
56    fn process_adjacent_segments(
57        &mut self,
58        active: Active<IMSegment<C>>,
59        other: &IMSegment<C>,
60    ) -> AdjProcOutput<C::Scalar> {
61        // NOTE: The below logic is a loop instead of a
62        // conditional due to FP issues. Specifically,
63        // sometimes, two non-overlapping lines may become
64        // overlapping once broken at the point of intersection!
65
66        // EXAMPLE:
67        // let pt_7 = Coord::from((-32.57812499999999, 241.33427773853316));
68        // let pt_8 = Coord::from((-36.11348070978957, 237.7989220287436));
69        // let pt_13 = Coord::from((-25.507080078124993, 248.40532266040816));
70        // let pt_14 = Coord::from((-36.48784219165816, 237.424560546875));
71        // let pt_16 = Coord::from((-36.048578439260666, 237.8638242992725));
72        // 7-8 and 13-14 intersect at 16 such that 8-16 and 14-16 overlap !
73
74        // We handle this by intersecting twice if the segments overlap after adjustment.
75        let mut out = AdjProcOutput {
76            isec: None,
77            should_continue: true,
78            should_callback: false,
79        };
80        while let Some(isec) = other.geom().intersect_line_ordered(&active.geom()) {
81            trace!("Found intersection (LL):\n\tsegment1: {:?}\n\tsegment2: {:?}\n\tintersection: {:?}", other, active, isec);
82            out.isec = Some(isec);
83
84            // 1. Split adj_segment, and extra splits to storage
85            let adj_overlap = active.adjust_one_segment(isec, |e| self.events.push(e));
86
87            // 2. Split segment, adding extra segments as needed.
88            let seg_overlap = other.adjust_one_segment(isec, |e| self.events.push(e));
89
90            assert_eq!(
91                adj_overlap.is_some(),
92                seg_overlap.is_some(),
93                "one of the intersecting segments had an overlap, but not the other!"
94            );
95            if let Some(adj_ovl) = adj_overlap {
96                let tgt = seg_overlap.unwrap();
97                trace!("setting overlap: {adj_ovl:?} -> {tgt:?}");
98                adj_ovl.chain_overlap(tgt.clone());
99
100                if &tgt == other {
101                    // The whole event segment is now overlapping
102                    // some other active segment.
103                    //
104                    // We do not need to continue iteration, but
105                    // should callback if the left event of the
106                    // now-parent has already been processed.
107                    out.should_callback = adj_ovl.is_left_event_done();
108                    out.should_continue = false;
109                }
110
111                // Overlaps are exact compute, so we do not need
112                // to re-run the loop.
113                return out;
114            }
115
116            if active.geom().partial_cmp(&other.geom()) == Some(Ordering::Equal) {
117                continue;
118            } else {
119                break;
120            }
121        }
122        out
123    }

125    fn handle_event<F>(&mut self, event: Event<C::Scalar, IMSegment<C>>, cb: &mut F) -> bool
126    where
127        F: for<'a> FnMut(&'a IMSegment<C>, EventType),
128    {
129        use EventType::*;
130        let segment = match IMSegment::is_correct(&event) {
131            false => return false,
132            _ => event.payload,
133        };
134        trace!(
135            "handling event: {pt:?} ({ty:?}) @ {seg:?}",
136            pt = event.point,
137            ty = event.ty,
138            seg = segment,
139        );
140
141        // let prev = self.active_segments.previous(&segment).cloned();
142        // let next = self.active_segments.next(&segment).cloned();
143
144        match &event.ty {
145            LineLeft => {
146                let mut should_add = true;
147                let mut insert_idx = self.active_segments.index_not_of(&segment);
148                if !self.is_simple {
149                    for is_next in [true, false].into_iter() {
150                        let active = if is_next {
151                            if insert_idx < self.active_segments.len() {
152                                self.active_segments[insert_idx].clone()
153                            } else {
154                                continue;
155                            }
156                        } else if insert_idx > 0 {
157                            self.active_segments[insert_idx - 1].clone()
158                        } else {
159                            continue;
160                        };
161                        let AdjProcOutput {
162                            isec,
163                            should_continue,
164                            should_callback,
165                        } = self.process_adjacent_segments(active.clone(), &segment);
166                        let isec = match isec {
167                            Some(isec) => isec,
168                            None => continue,
169                        };
170                        // A special case is if adj_segment was split, and the
171                        // intersection is at the start of this segment. In this
172                        // case, there is an right-end event in the heap, that
173                        // needs to be handled before finishing up this event.
174                        let handle_end_event = {
175                            // Get first point of intersection
176                            let int_pt = isec.left();
177                            // Check its not first point of the adjusted, but is
178                            // first point of current segment
179                            int_pt != active.geom().left() && int_pt == segment.geom().left()
180                        };
181                        if handle_end_event {
182                            let event = self.events.pop().unwrap();
183                            let done = self.handle_event(event, cb);
184                            debug_assert!(done, "special right-end event handling failed");
185                            if !is_next {
186                                // The prev-segment is now removed
187                                insert_idx -= 1;
188                            }
189                        }
190
191                        if !should_continue {
192                            should_add = false;
193                            if !should_callback {
194                                return true;
195                            }
196                            break;
197                        }
198
199                        // let n = self.active_segments.len();
200                        // if is_next && 1 + insert_idx < n {
201                        //     (insert_idx..n).find(|&idx| !self.active_segments.check_swap(idx));
202                        // } else if !is_next && insert_idx > 1 {
203                        //     (0..insert_idx - 2)
204                        //         .rev()
205                        //         .find(|&idx| !self.active_segments.check_swap(idx));
206                        // }
207                    }
208                }
209
210                if should_add {
211                    // Add current segment as active
212                    // Safety: `self.segments` is a `Box` that is not
213                    // de-allocated until `self` is dropped.
214                    debug!("insert_active: {segment:?}");
215
216                    // NOTE: we bravely track insert_idx as the active-list is adjusted
217                    // self.active_segments.insert_active(segment.clone());
218                    self.active_segments.insert_at(insert_idx, segment.clone());
219                }
220
221                let mut cb_seg = Some(segment);
222                while let Some(seg) = cb_seg {
223                    cb(&seg, event.ty);
224                    seg.set_left_event_done();
225                    cb_seg = seg.overlap();
226                }
227            }
228            LineRight => {
229                // Safety: `self.segments` is a `Box` that is not
230                // de-allocated until `self` is dropped.
231                debug!("remove_active: {segment:?}");
232                let el_idx = self.active_segments.index_of(&segment);
233                let prev = (el_idx > 0).then(|| self.active_segments[el_idx - 1].clone());
234                let next = (1 + el_idx < self.active_segments.len())
235                    .then(|| self.active_segments[el_idx + 1].clone());
236                assert_eq!(self.active_segments.remove_at(el_idx), segment);
237
238                let mut cb_seg = Some(segment);
239                while let Some(seg) = cb_seg {
240                    cb(&seg, event.ty);
241                    cb_seg = seg.overlap();
242                }
243
244                if !self.is_simple {
245                    if let (Some(prev), Some(next)) = (prev, next) {
246                        let prev_geom = prev.geom();
247                        let next_geom = next.geom();
248                        if let Some(adj_intersection) = prev_geom.intersect_line_ordered(&next_geom)
249                        {
250                            // 1. Split prev_segment, and extra splits to storage
251                            let first = prev
252                                .adjust_one_segment(adj_intersection, |e| self.events.push(e))
253                                .is_none();
254                            let second = next
255                                .adjust_one_segment(adj_intersection, |e| self.events.push(e))
256                                .is_none();
257                            debug_assert!(
258                                first && second,
259                                "adjacent segments @ removal can't overlap!"
260                            );
261                        }
262                    }
263                }
264            }
265            PointLeft => {
266                if !self.is_simple {
267                    let insert_idx = self.active_segments.index_not_of(&segment);
268                    let prev =
269                        (insert_idx > 0).then(|| self.active_segments[insert_idx - 1].clone());
270                    let next = (insert_idx < self.active_segments.len())
271                        .then(|| self.active_segments[insert_idx].clone());
272
273                    for adj_segment in prev.into_iter().chain(next.into_iter()) {
274                        let geom = adj_segment.geom();
275                        if let Some(adj_intersection) = segment.geom().intersect_line_ordered(&geom)
276                        {
277                            trace!("Found intersection (PL):\n\tsegment1: {:?}\n\tsegment2: {:?}\n\tintersection: {:?}", segment, adj_segment, adj_intersection);
278                            // 1. Split adj_segment, and extra splits to storage
279                            let adj_overlap = adj_segment
280                                .adjust_one_segment(adj_intersection, |e| self.events.push(e));
281
282                            // Can't have overlap with a point
283                            debug_assert!(adj_overlap.is_none());
284                        }
285                    }
286                }
287
288                // Points need not be active segments.
289                // Send the point-segment to callback.
290                cb(&segment, event.ty);
291            }
292            PointRight => {
293                // Nothing to do. We could remove this variant once we
294                // are confident about the logic.
295            }
296        }
297        true
298    }
