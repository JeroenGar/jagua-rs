8/// An ordered list of [`EdgeEndBundle`]s around a [`RelateNodeFactory::Node`].

32    pub(crate) fn new(
33        edges: Vec<LabeledEdgeEndBundle<F>>,
34        graph_a: &GeometryGraph<F>,
35        graph_b: &GeometryGraph<F>,
36    ) -> Self {
37        let mut labeled_bundle_star = Self { edges };
38        labeled_bundle_star.compute_labeling(graph_a, graph_b);
39        labeled_bundle_star
40    }

42    /// Compute a label for the star based on the labels of its EdgeEndBundles.
43    fn compute_labeling(&mut self, graph_a: &GeometryGraph<F>, graph_b: &GeometryGraph<F>) {
44        self.propagate_side_labels(0);
45        self.propagate_side_labels(1);
46        let mut has_dimensional_collapse_edge = [false, false];
47        for edge_end in self.edges.iter() {
48            let label = edge_end.label();
49            for (geom_index, is_collapsed) in has_dimensional_collapse_edge.iter_mut().enumerate() {
50                *is_collapsed = label.is_line(geom_index)
51                    && label.on_position(geom_index) == Some(CoordPos::OnBoundary);
52            }
53        }
54        for edge_end_bundle in &mut self.edges {
55            let coord = *edge_end_bundle.coordinate();
56            let label = edge_end_bundle.label_mut();
57            for (geom_index, is_dimensionally_collapsed) in
58                has_dimensional_collapse_edge.iter().enumerate()
59            {
60                if label.is_any_empty(geom_index) {
61                    let position: CoordPos = if *is_dimensionally_collapsed {
62                        CoordPos::Outside
63                    } else {
64                        // PERF: In JTS this is memoized, but that gets a little tricky with rust's
65                        // borrow checker. Let's wait to see if it's a hotspot.
66                        let geometry = match geom_index {
67                            0 => graph_a.geometry(),
68                            1 => graph_b.geometry(),
69                            _ => unreachable!("invalid geom_index"),
70                        };
71                        use crate::HasDimensions;
72                        if geometry.dimensions() == Dimensions::TwoDimensional {
73                            geometry.coordinate_position(&coord)
74                        } else {
75                            // if geometry is *not* an area, Coord is always Outside
76                            CoordPos::Outside
77                        }
78                    };
79                    label.set_all_positions_if_empty(geom_index, position);
80                }
81            }
82        }
83        debug!("edge_end_bundle_star: {:?}", self);
84    }
85
86    fn propagate_side_labels(&mut self, geom_index: usize) {
87        let mut start_position = None;
88
89        for edge_ends in self.edge_end_bundles_iter() {
90            let label = edge_ends.label();
91            if label.is_geom_area(geom_index) {
92                if let Some(position) = label.position(geom_index, Direction::Left) {
93                    start_position = Some(position);
94                }
95            }
96        }
97        if start_position.is_none() {
98            return;
99        }
100        let mut current_position = start_position.unwrap();
101
102        for edge_ends in self.edge_end_bundles_iter_mut() {
103            let label = edge_ends.label_mut();
104            if label.position(geom_index, Direction::On).is_none() {
105                label.set_position(geom_index, Direction::On, current_position);
106            }
107            if label.is_geom_area(geom_index) {
108                let left_position = label.position(geom_index, Direction::Left);
109                let right_position = label.position(geom_index, Direction::Right);
110
111                if let Some(right_position) = right_position {
112                    debug_assert!(right_position == current_position, "side_location conflict with coordinate: {:?}, right_location: {:?}, current_location: {:?}", edge_ends.coordinate(), right_position, current_position);
113                    assert!(left_position.is_some(), "found single null side");
114                    current_position = left_position.unwrap();
115                } else {
116                    debug_assert!(label.position(geom_index, Direction::Left).is_none());
117                    label.set_position(geom_index, Direction::Right, current_position);
118                    label.set_position(geom_index, Direction::Left, current_position);
119                }
120            }
121        }
122    }

132    pub fn update_intersection_matrix(&self, intersection_matrix: &mut IntersectionMatrix) {
133        for edge_end_bundle in self.edge_end_bundles_iter() {
134            edge_end_bundle.update_intersection_matrix(intersection_matrix);
135            debug!(
136                "updated intersection_matrix: {:?} from edge_end_bundle: {:?}",
137                intersection_matrix, edge_end_bundle
138            );
139        }
140    }

143impl<F> EdgeEndBundleStar<F>
144where

154    pub(crate) fn insert(&mut self, edge_end: EdgeEnd<F>) {
155        let bundle = self
156            .edge_map
157            .entry(edge_end.key().clone())
158            .or_insert_with(|| EdgeEndBundle::new(*edge_end.coordinate()));
159        bundle.insert(edge_end);
160    }
161
162    fn edge_end_bundles_iter(&self) -> impl Iterator<Item = &EdgeEndBundle<F>> {
163        self.edge_map.values()
164    }

166    fn edge_end_bundles_iter_mut(&mut self) -> impl Iterator<Item = &mut EdgeEndBundle<F>> {
167        self.edge_map.values_mut()
168    }

170    /// Compute labeling for the star's EdgeEndBundles, and use them to compute an overall label
171    /// for the star.
172    ///
173    /// Implementation Note: This is a bit of a divergence from JTS in two ways.
174    ///
175    /// Firstly, JTS doesn't leverage optionals, and sets nullable `Label`s, whereas here we convert
176    /// to an explicitly labeled type to avoid unwrapping optionals later.
177    ///
178    /// Secondly, in JTS this functionality does not live directly on EdgeEndBundleStar, but rather
179    /// on it's parent class [EdgeEndStar](https://github.com/locationtech/jts/blob/jts-1.18.1/modules/core/src/main/java/org/locationtech/jts/geomgraph/EdgeEndStar.java#L117)
180    ///
181    /// Since we're only using this one subclass (EdgeEndBundleStar), we skip the
182    /// complexity of mapping Java inheritance to Rust and implement this functionality
183    /// on EdgeEndBundleStar directly.
184    ///
185    /// If/When we implement overlay operations we might consider extracting the superclass
186    /// behavior.
187    pub(crate) fn into_labeled(
188        self,
189        graph_a: &GeometryGraph<F>,
190        graph_b: &GeometryGraph<F>,
191    ) -> LabeledEdgeEndBundleStar<F> {
192        debug!("edge_end_bundle_star: {:?}", self);
193        let labeled_edges = self
194            .edge_map
195            .into_values()
196            .map(|edge_end_bundle| edge_end_bundle.into_labeled())
197            .collect();
198        LabeledEdgeEndBundleStar::new(labeled_edges, graph_a, graph_b)
199    }
