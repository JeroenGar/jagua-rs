7/// An `Edge` represents a one dimensional line in a geometry.
8///
9/// This is based on [JTS's `Edge` as of 1.18.1](https://github.com/locationtech/jts/blob/jts-1.18.1/modules/core/src/main/java/org/locationtech/jts/geomgraph/Edge.java)
10#[derive(Debug)]
11pub(crate) struct Edge<F: GeoFloat> {
12    /// `coordinates` of the line geometry
13    coords: Vec<Coord<F>>,
14
15    /// an edge is "isolated" if no other edge touches it
16    is_isolated: bool,
17
18    /// other edges that this edge intersects with
19    edge_intersections: BTreeSet<EdgeIntersection<F>>,
20
21    /// where the line's topological classification to the two geometries is recorded
22    label: Label,
23}
24
25impl<F: GeoFloat> Edge<F> {
26    /// Create a new Edge.
27    ///
28    /// - `coords` a *non-empty* Vec of Coords
29    /// - `label` an appropriately dimensioned topology label for the Edge. See [`TopologyPosition`]
30    ///    for details
31    pub(crate) fn new(mut coords: Vec<Coord<F>>, label: Label) -> Edge<F> {
32        assert!(!coords.is_empty(), "Can't add empty edge");
33        // Once set, `edge.coords` never changes length.
34        coords.shrink_to_fit();
35        Edge {
36            coords,
37            label,
38            is_isolated: true,
39            edge_intersections: BTreeSet::new(),
40        }
41    }
42
43    pub(crate) fn label(&self) -> &Label {
44        &self.label
45    }
46
47    pub(crate) fn label_mut(&mut self) -> &mut Label {
48        &mut self.label
49    }
50
51    pub fn coords(&self) -> &[Coord<F>] {
52        &self.coords
53    }
54
55    pub fn is_isolated(&self) -> bool {
56        self.is_isolated
57    }
58    pub fn mark_as_unisolated(&mut self) {
59        self.is_isolated = false;
60    }
61
62    pub fn edge_intersections(&self) -> &BTreeSet<EdgeIntersection<F>> {
63        &self.edge_intersections
64    }
65
66    pub fn edge_intersections_mut(&mut self) -> &mut BTreeSet<EdgeIntersection<F>> {
67        &mut self.edge_intersections
68    }
69
70    pub fn add_edge_intersection_list_endpoints(&mut self) {
71        let max_segment_index = self.coords().len() - 1;
72        let first_coord = self.coords()[0];
73        let max_coord = self.coords()[max_segment_index];
74        self.edge_intersections_mut()
75            .insert(EdgeIntersection::new(first_coord, 0, F::zero()));
76        self.edge_intersections_mut().insert(EdgeIntersection::new(
77            max_coord,
78            max_segment_index,
79            F::zero(),
80        ));
81    }
82
83    pub fn is_closed(&self) -> bool {
84        self.coords().first() == self.coords().last()
85    }
86
87    /// Adds EdgeIntersections for one or both intersections found for a segment of an edge to the
88    /// edge intersection list.
89    pub fn add_intersections(
90        &mut self,
91        intersection: LineIntersection<F>,
92        line: Line<F>,
93        segment_index: usize,
94    ) {
95        match intersection {
96            LineIntersection::SinglePoint { intersection, .. } => {
97                self.add_intersection(intersection, line, segment_index);
98            }
99            LineIntersection::Collinear { intersection } => {
100                self.add_intersection(intersection.start, line, segment_index);
101                self.add_intersection(intersection.end, line, segment_index);
102            }
103        }
104    }
105
106    /// Add an EdgeIntersection for `intersection`.
107    ///
108    /// An intersection that falls exactly on a vertex of the edge is normalized to use the higher
109    /// of the two possible `segment_index`
110    pub fn add_intersection(
111        &mut self,
112        intersection_coord: Coord<F>,
113        line: Line<F>,
114        segment_index: usize,
115    ) {
116        let mut normalized_segment_index = segment_index;
117        let mut distance = RobustLineIntersector::compute_edge_distance(intersection_coord, line);
118
119        let next_segment_index = normalized_segment_index + 1;
120
121        if next_segment_index < self.coords.len() {
122            let next_coord = self.coords[next_segment_index];
123            if intersection_coord == next_coord {
124                normalized_segment_index = next_segment_index;
125                distance = F::zero();
126            }
127        }
128        self.edge_intersections.insert(EdgeIntersection::new(
129            intersection_coord,
130            normalized_segment_index,
131            distance,
132        ));
133    }
134
135    /// Update the IM with the contribution for this component.
136    ///
137    /// A component only contributes if it has a labelling for both parent geometries
138    pub fn update_intersection_matrix(label: &Label, intersection_matrix: &mut IntersectionMatrix) {
139        intersection_matrix.set_at_least_if_in_both(
140            label.position(0, Direction::On),
141            label.position(1, Direction::On),
142            Dimensions::OneDimensional,
143        );
144
145        if label.is_area() {
146            intersection_matrix.set_at_least_if_in_both(
147                label.position(0, Direction::Left),
148                label.position(1, Direction::Left),
149                Dimensions::TwoDimensional,
150            );
151            intersection_matrix.set_at_least_if_in_both(
152                label.position(0, Direction::Right),
153                label.position(1, Direction::Right),
154                Dimensions::TwoDimensional,
155            );
156        }
157    }
