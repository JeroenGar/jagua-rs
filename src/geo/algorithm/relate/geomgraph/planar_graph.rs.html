7use std::cell::RefCell;

10pub(crate) struct PlanarGraphNode;
11
12/// The basic node constructor does not allow for incident edges
13impl<F> NodeFactory<F> for PlanarGraphNode
14where
15    F: GeoFloat,
16{
17    type Node = CoordNode<F>;
18    fn create_node(coordinate: Coord<F>) -> Self::Node {
19        CoordNode::new(coordinate)
20    }
21}
22
23pub(crate) struct PlanarGraph<F: GeoFloat> {
24    pub(crate) nodes: NodeMap<F, PlanarGraphNode>,
25    edges: Vec<Rc<RefCell<Edge<F>>>>,
26}
27
28impl<F: GeoFloat> PlanarGraph<F> {
29    pub fn edges(&self) -> &[Rc<RefCell<Edge<F>>>] {
30        &self.edges
31    }
32
33    pub fn new() -> Self {
34        PlanarGraph {
35            nodes: NodeMap::new(),
36            edges: vec![],
37        }
38    }
39
40    pub fn is_boundary_node(&self, geom_index: usize, coord: Coord<F>) -> bool {
41        self.nodes
42            .find(coord)
43            .and_then(|node| node.label().on_position(geom_index))
44            .map(|position| position == CoordPos::OnBoundary)
45            .unwrap_or(false)
46    }
47
48    pub fn insert_edge(&mut self, edge: Edge<F>) {
49        self.edges.push(Rc::new(RefCell::new(edge)));
50    }
51
52    pub fn add_node_with_coordinate(&mut self, coord: Coord<F>) -> &mut CoordNode<F> {
53        self.nodes.insert_node_with_coordinate(coord)
54    }
55
56    pub fn boundary_nodes(&self, geom_index: usize) -> impl Iterator<Item = &CoordNode<F>> {
57        self.nodes.iter().filter(move |node| {
58            matches!(
59                node.label().on_position(geom_index),
60                Some(CoordPos::OnBoundary)
61            )
62        })
63    }
64}
