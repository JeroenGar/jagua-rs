15/// Computes an [`IntersectionMatrix`] describing the topological relationship between two
16/// Geometries.
17///
18/// `RelateOperation` does not currently support [`GeometryCollection`]s with overlapping Polygons,
19/// and may provide surprising results in that case.
20///
21/// This implementation relies heavily on the functionality of [`GeometryGraph`].
22///
23/// Based on [JTS's `RelateComputer` as of 1.18.1](https://github.com/locationtech/jts/blob/jts-1.18.1/modules/core/src/main/java/org/locationtech/jts/operation/relate/RelateComputer.java)
24pub(crate) struct RelateOperation<'a, F>
25where
26    F: GeoFloat,
27{
28    graph_a: GeometryGraph<'a, F>,
29    graph_b: GeometryGraph<'a, F>,
30    nodes: NodeMap<F, RelateNodeFactory>,
31    line_intersector: RobustLineIntersector,
32    isolated_edges: Vec<Rc<RefCell<Edge<F>>>>,
33}
34

36impl<F> NodeFactory<F> for RelateNodeFactory
37where

40    type Node = (CoordNode<F>, EdgeEndBundleStar<F>);
41    fn create_node(coordinate: Coord<F>) -> Self::Node {
42        (CoordNode::new(coordinate), EdgeEndBundleStar::new())
43    }
44}

50    pub(crate) fn new(
51        geom_a: &'a GeometryCow<'a, F>,
52        geom_b: &'a GeometryCow<'a, F>,
53    ) -> RelateOperation<'a, F> {
54        Self {
55            graph_a: GeometryGraph::new(0, geom_a),
56            graph_b: GeometryGraph::new(1, geom_b),
57            nodes: NodeMap::new(),
58            isolated_edges: vec![],
59            line_intersector: RobustLineIntersector::new(),
60        }
61    }

63    pub(crate) fn compute_intersection_matrix(&mut self) -> IntersectionMatrix {
64        let mut intersection_matrix = IntersectionMatrix::empty();
65        // since Geometries are finite and embedded in a 2-D space,
66        // the `(Outside, Outside)` element must always be 2-D
67        intersection_matrix.set(
68            CoordPos::Outside,
69            CoordPos::Outside,
70            Dimensions::TwoDimensional,
71        );
72
73        use crate::BoundingRect;
74        use crate::Intersects;
75        match (
76            self.graph_a.geometry().bounding_rect(),
77            self.graph_b.geometry().bounding_rect(),
78        ) {
79            (Some(bounding_rect_a), Some(bounding_rect_b))
80                if bounding_rect_a.intersects(&bounding_rect_b) => {}
81            _ => {
82                // since Geometries don't overlap, we can skip most of the work
83                self.compute_disjoint_intersection_matrix(&mut intersection_matrix);
84                return intersection_matrix;
85            }
86        }
87
88        // Since changes to topology are inspected at nodes, we must crate a node for each
89        // intersection.
90        self.graph_a
91            .compute_self_nodes(Box::new(self.line_intersector.clone()));
92        self.graph_b
93            .compute_self_nodes(Box::new(self.line_intersector.clone()));
94
95        // compute intersections between edges of the two input geometries
96        let segment_intersector = self
97            .graph_a
98            .compute_edge_intersections(&self.graph_b, Box::new(self.line_intersector.clone()));
99
100        self.compute_intersection_nodes(0);
101        self.compute_intersection_nodes(1);
102        // Copy the labelling for the nodes in the parent Geometries.  These override any labels
103        // determined by intersections between the geometries.
104        self.copy_nodes_and_labels(0);
105        self.copy_nodes_and_labels(1);
106        // complete the labelling for any nodes which only have a label for a single geometry
107        self.label_isolated_nodes();
108        // If a proper intersection was found, we can set a lower bound on the IM.
109        self.compute_proper_intersection_im(&segment_intersector, &mut intersection_matrix);
110        // Now process improper intersections
111        // (eg where one or other of the geometries has a vertex at the intersection point)
112        // We need to compute the edge graph at all nodes to determine the IM.
113        let edge_end_builder = EdgeEndBuilder::new();
114        let edge_ends_a: Vec<_> = edge_end_builder.compute_ends_for_edges(self.graph_a.edges());
115        self.insert_edge_ends(edge_ends_a);
116        let edge_ends_b: Vec<_> = edge_end_builder.compute_ends_for_edges(self.graph_b.edges());
117        self.insert_edge_ends(edge_ends_b);
118
119        let mut nodes = NodeMap::new();
120        std::mem::swap(&mut self.nodes, &mut nodes);
121        let labeled_node_edges = nodes
122            .into_iter()
123            .map(|(node, edges)| (node, edges.into_labeled(&self.graph_a, &self.graph_b)))
124            .collect();
125
126        // Compute the labeling for "isolated" components
127        //
128        // Isolated components are components that do not touch any other components in the graph.
129        //
130        // They can be identified by the fact that their labels will have only one non-empty
131        // element, the one for their parent geometry.
132        //
133        // We only need to check components contained in the input graphs, since, by definition,
134        // isolated components will not have been replaced by new components formed by
135        // intersections.
136        self.label_isolated_edges(0, 1);
137        self.label_isolated_edges(1, 0);
138
139        debug!(
140            "before update_intersection_matrix: {:?}",
141            &intersection_matrix
142        );
143        self.update_intersection_matrix(labeled_node_edges, &mut intersection_matrix);
144
145        intersection_matrix
146    }

148    fn insert_edge_ends(&mut self, edge_ends: Vec<EdgeEnd<F>>) {
149        for edge_end in edge_ends {
150            let (_node, edges) = self
151                .nodes
152                .insert_node_with_coordinate(*edge_end.coordinate());
153            edges.insert(edge_end);
154        }
155    }

157    fn compute_proper_intersection_im(
158        &mut self,
159        segment_intersector: &SegmentIntersector<F>,
160        intersection_matrix: &mut IntersectionMatrix,
161    ) {
162        // If a proper intersection is found, we can set a lower bound on the IM.
163        let dim_a = self.graph_a.geometry().dimensions();
164        let dim_b = self.graph_b.geometry().dimensions();
165
166        let has_proper = segment_intersector.has_proper_intersection();
167        let has_proper_interior = segment_intersector.has_proper_interior_intersection();
168
169        debug_assert!(
170            (dim_a != Dimensions::ZeroDimensional && dim_b != Dimensions::ZeroDimensional)
171                || (!has_proper && !has_proper_interior)
172        );
173
174        match (dim_a, dim_b) {
175            // If edge segments of Areas properly intersect, the areas must properly overlap.
176            (Dimensions::TwoDimensional, Dimensions::TwoDimensional) => {
177                if has_proper {
178                    intersection_matrix
179                        .set_at_least_from_string("212101212")
180                        .expect("error in hardcoded dimensions");
181                }
182            }
183
184            // If a Line segment properly intersects an edge segment of an Area, it follows that
185            // the Interior of the Line intersects the Boundary of the Area.
186            // If the intersection is a proper *interior* intersection, then there is an
187            // Interior-Interior intersection too.
188            // Note that it does not follow that the Interior of the Line intersects the Exterior
189            // of the Area, since there may be another Area component which contains the rest of the Line.
190            (Dimensions::TwoDimensional, Dimensions::OneDimensional) => {
191                if has_proper {
192                    intersection_matrix
193                        .set_at_least_from_string("FFF0FFFF2")
194                        .expect("error in hardcoded dimensions");
195                }
196
197                if has_proper_interior {
198                    intersection_matrix
199                        .set_at_least_from_string("1FFFFF1FF")
200                        .expect("error in hardcoded dimensions");
201                }
202            }
203
204            (Dimensions::OneDimensional, Dimensions::TwoDimensional) => {
205                if has_proper {
206                    intersection_matrix
207                        .set_at_least_from_string("F0FFFFFF2")
208                        .expect("error in hardcoded dimensions");
209                }
210
211                if has_proper_interior {
212                    intersection_matrix
213                        .set_at_least_from_string("1F1FFFFFF")
214                        .expect("error in hardcoded dimensions");
215                }
216            }
217
218            // If edges of LineStrings properly intersect *in an interior point*, all we can deduce
219            // is that the interiors intersect.  (We can NOT deduce that the exteriors intersect,
220            // since some other segments in the geometries might cover the points in the
221            // neighbourhood of the intersection.)
222            // It is important that the point be known to be an interior point of both Geometries,
223            // since it is possible in a self-intersecting geometry to have a proper intersection
224            // on one segment that is also a boundary point of another segment.
225            (Dimensions::OneDimensional, Dimensions::OneDimensional) => {
226                if has_proper_interior {
227                    intersection_matrix
228                        .set_at_least_from_string("0FFFFFFFF")
229                        .expect("error in hardcoded dimensions");
230                }
231            }
232            _ => {}
233        }
234    }

236    /// Copy all nodes from an arg geometry into this graph.
237    ///
238    /// The node label in the arg geometry overrides any previously computed label for that
239    /// argIndex.  (E.g. a node may be an intersection node with a computed label of BOUNDARY, but
240    /// in the original arg Geometry it is actually in the interior due to the Boundary
241    /// Determination Rule)
242    fn copy_nodes_and_labels(&mut self, geom_index: usize) {
243        let graph = if geom_index == 0 {
244            &self.graph_a
245        } else {
246            assert_eq!(geom_index, 1);
247            &self.graph_b
248        };
249        for graph_node in graph.nodes_iter() {
250            let new_node = self
251                .nodes
252                .insert_node_with_coordinate(*graph_node.coordinate());
253
254            let on_position = graph_node
255                .label()
256                .on_position(geom_index)
257                .expect("node should have been labeled by now");
258
259            new_node.0.set_label_on_position(geom_index, on_position);
260        }
261    }
262
263    /// Insert nodes for all intersections on the edges of a Geometry.
264    ///
265    /// Label the created nodes the same as the edge label if they do not already have a label.
266    /// This allows nodes created by either self-intersections or mutual intersections to be
267    /// labeled.
268    ///
269    /// Endpoint nodes will already be labeled from when they were inserted.
270    fn compute_intersection_nodes(&mut self, geom_index: usize) {
271        let graph = if geom_index == 0 {
272            &self.graph_a
273        } else {
274            assert_eq!(geom_index, 1);
275            &self.graph_b
276        };
277
278        for edge in graph.edges() {
279            let edge = edge.borrow();
280
281            let edge_position = edge.label().on_position(geom_index);
282            for edge_intersection in edge.edge_intersections() {
283                let (new_node, _edges) = self
284                    .nodes
285                    .insert_node_with_coordinate(edge_intersection.coordinate());
286
287                if edge_position == Some(CoordPos::OnBoundary) {
288                    new_node.set_label_boundary(geom_index);
289                } else if new_node.label().is_empty(geom_index) {
290                    new_node.set_label_on_position(geom_index, CoordPos::Inside);
291                }
292            }
293        }
294    }

296    /// If the Geometries are disjoint, we need to enter their dimension and boundary dimension in
297    /// the `Outside` rows in the IM
298    fn compute_disjoint_intersection_matrix(&self, intersection_matrix: &mut IntersectionMatrix) {
299        {
300            let geometry_a = self.graph_a.geometry();
301            let dimensions = geometry_a.dimensions();
302            if dimensions != Dimensions::Empty {
303                intersection_matrix.set(CoordPos::Inside, CoordPos::Outside, dimensions);
304
305                let boundary_dimensions = geometry_a.boundary_dimensions();
306                if boundary_dimensions != Dimensions::Empty {
307                    intersection_matrix.set(
308                        CoordPos::OnBoundary,
309                        CoordPos::Outside,
310                        boundary_dimensions,
311                    );
312                }
313            }
314        }
315
316        {
317            let geometry_b = self.graph_b.geometry();
318            let dimensions = geometry_b.dimensions();
319            if dimensions != Dimensions::Empty {
320                intersection_matrix.set(CoordPos::Outside, CoordPos::Inside, dimensions);
321
322                let boundary_dimensions = geometry_b.boundary_dimensions();
323                if boundary_dimensions != Dimensions::Empty {
324                    intersection_matrix.set(
325                        CoordPos::Outside,
326                        CoordPos::OnBoundary,
327                        boundary_dimensions,
328                    );
329                }
330            }
331        }
332    }

334    fn update_intersection_matrix(
335        &self,
336        labeled_node_edges: Vec<(CoordNode<F>, LabeledEdgeEndBundleStar<F>)>,
337        intersection_matrix: &mut IntersectionMatrix,
338    ) {
339        debug!(
340            "before updated_intersection_matrix(isolated_edges): {:?}",
341            intersection_matrix
342        );
343        for isolated_edge in &self.isolated_edges {
344            let edge = isolated_edge.borrow();
345            Edge::<F>::update_intersection_matrix(edge.label(), intersection_matrix);
346            debug!(
347                "after isolated_edge update_intersection_matrix: {:?}, (isolated_edge: {:?}, label: {:?})",
348                intersection_matrix,
349                edge,
350                edge.label()
351            );
352        }
353
354        for (node, edges) in labeled_node_edges.iter() {
355            node.update_intersection_matrix(intersection_matrix);
356            edges.update_intersection_matrix(intersection_matrix);
357        }
358    }

360    /// Processes isolated edges by computing their labelling and adding them to the isolated edges
361    /// list.
362    ///
363    /// By definition, "isolated" edges are guaranteed not to touch the boundary of the target
364    /// (since if they did, they would have caused an intersection to be computed and hence would
365    /// not be isolated).
366    fn label_isolated_edges(&mut self, this_index: usize, target_index: usize) {
367        let (this_graph, target_graph) = if this_index == 0 {
368            (&self.graph_a, &self.graph_b)
369        } else {
370            (&self.graph_b, &self.graph_a)
371        };
372
373        for edge in this_graph.edges() {
374            let mut mut_edge = edge.borrow_mut();
375            if mut_edge.is_isolated() {
376                Self::label_isolated_edge(&mut mut_edge, target_index, target_graph.geometry());
377                self.isolated_edges.push(edge.clone());
378            }
379        }
380    }
381
382    /// Label an isolated edge of a graph with its relationship to the target geometry.
383    /// If the target has dim 2 or 1, the edge can either be in the interior or the exterior.
384    /// If the target has dim 0, the edge must be in the exterior
385    fn label_isolated_edge(edge: &mut Edge<F>, target_index: usize, target: &GeometryCow<F>) {
386        if target.dimensions() > Dimensions::ZeroDimensional {
387            // An isolated edge doesn't cross any boundary, so it's either wholly inside, or wholly
388            // outside of the geometry. As such, we can use any point from the edge to infer the
389            // position of the edge as a whole.
390            let coord = edge.coords().first().expect("can't create empty edge");
391            let position = target.coordinate_position(coord);
392            edge.label_mut().set_all_positions(target_index, position);
393        } else {
394            edge.label_mut()
395                .set_all_positions(target_index, CoordPos::Outside);
396        }
397    }

399    /// Isolated nodes are nodes whose labels are incomplete (e.g. the location for one Geometry is
400    /// null).
401    /// This is the case because nodes in one graph which don't intersect nodes in the other
402    /// are not completely labeled by the initial process of adding nodes to the nodeList.  To
403    /// complete the labelling we need to check for nodes that lie in the interior of edges, and in
404    /// the interior of areas.
405    fn label_isolated_nodes(&mut self) {
406        let geometry_a = self.graph_a.geometry();
407        let geometry_b = self.graph_b.geometry();
408        for (node, _edges) in self.nodes.iter_mut() {
409            let label = node.label();
410            // isolated nodes should always have at least one geometry in their label
411            debug_assert!(label.geometry_count() > 0, "node with empty label found");
412            if node.is_isolated() {
413                if label.is_empty(0) {
414                    Self::label_isolated_node(node, 0, geometry_a)
415                } else {
416                    Self::label_isolated_node(node, 1, geometry_b)
417                }
418            }
419        }
420    }
421
422    fn label_isolated_node(
423        node: &mut CoordNode<F>,
424        target_index: usize,
425        geometry: &GeometryCow<F>,
426    ) {
427        let position = geometry.coordinate_position(node.coordinate());
428        node.label_mut().set_all_positions(target_index, position);
429    }

439    fn test_disjoint() {
440        let square_a: Geometry = polygon![
441            (x: 0., y: 0.),
442            (x: 0., y: 20.),
443            (x: 20., y: 20.),
444            (x: 20., y: 0.),
445            (x: 0., y: 0.),

449        let square_b: Geometry = polygon![
450            (x: 55., y: 55.),
451            (x: 50., y: 60.),
452            (x: 60., y: 60.),
453            (x: 60., y: 55.),
454            (x: 55., y: 55.),
455        ]
456        .into();
457
458        let gc1 = GeometryCow::from(&square_a);
459        let gc2 = GeometryCow::from(&square_b);
460        let mut relate_computer = RelateOperation::new(&gc1, &gc2);
