22pub fn get_bounding_rect<I, C, T>(collection: I) -> Option<Rect<T>>
23where
24    T: CoordNum,
25    C: AsRef<Coord<T>>,
26    I: IntoIterator<Item = C>,
27{
28    let mut iter = collection.into_iter();
29    if let Some(pnt) = iter.next() {
30        let pnt = pnt.as_ref();
31        let mut xrange = (pnt.x, pnt.x);
32        let mut yrange = (pnt.y, pnt.y);
33        for pnt in iter {
34            let (px, py) = pnt.as_ref().x_y();
35            xrange = get_min_max(px, xrange.0, xrange.1);
36            yrange = get_min_max(py, yrange.0, yrange.1);
37        }
38
39        return Some(Rect::new(
40            coord! {
41                x: xrange.0,
42                y: yrange.0,
43            },
44            coord! {
45                x: xrange.1,
46                y: yrange.1,
47            },
48        ));
49    }
50    None
51}

53fn get_min_max<T: PartialOrd>(p: T, min: T, max: T) -> (T, T) {
54    if p > max {
55        (min, p)
56    } else if p < min {
57        (p, max)
58    } else {
59        (min, max)
60    }
61}

63pub fn line_segment_distance<T, C>(point: C, start: C, end: C) -> T
64where
65    T: CoordFloat,
66    C: Into<Coord<T>>,
67{
68    let point = point.into();
69    let start = start.into();
70    let end = end.into();
71
72    if start == end {
73        return line_euclidean_length(Line::new(point, start));
74    }
75    let dx = end.x - start.x;
76    let dy = end.y - start.y;
77    let d_squared = dx * dx + dy * dy;
78    let r = ((point.x - start.x) * dx + (point.y - start.y) * dy) / d_squared;
79    if r <= T::zero() {
80        return line_euclidean_length(Line::new(point, start));
81    }
82    if r >= T::one() {
83        return line_euclidean_length(Line::new(point, end));
84    }
85    let s = ((start.y - point.y) * dx - (start.x - point.x) * dy) / d_squared;
86    s.abs() * dx.hypot(dy)
87}

89pub fn line_euclidean_length<T>(line: Line<T>) -> T
90where
91    T: CoordFloat,
92{
93    line.dx().hypot(line.dy())
94}

96pub fn point_line_string_euclidean_distance<T>(p: Point<T>, l: &LineString<T>) -> T
97where
98    T: CoordFloat,
99{
100    // No need to continue if the point is on the LineString, or it's empty
101    if line_string_contains_point(l, p) || l.0.is_empty() {
102        return T::zero();
103    }
104    l.lines()
105        .map(|line| line_segment_distance(p.0, line.start, line.end))
106        .fold(T::max_value(), |accum, val| accum.min(val))
107}

117pub fn point_contains_point<T>(p1: Point<T>, p2: Point<T>) -> bool

119    T: CoordFloat,

125pub fn line_string_contains_point<T>(line_string: &LineString<T>, point: Point<T>) -> bool
126where
127    T: CoordFloat,
128{
129    // LineString without points
130    if line_string.0.is_empty() {
131        return false;
132    }
133    // LineString with one point equal p
134    if line_string.0.len() == 1 {
135        return point_contains_point(Point::from(line_string[0]), point);
136    }
137    // check if point is a vertex
138    if line_string.0.contains(&point.0) {
139        return true;
140    }
141    for line in line_string.lines() {
142        // This is a duplicate of the line-contains-point logic in the "intersects" module
143        let tx = if line.dx() == T::zero() {
144            None
145        } else {
146            Some((point.x() - line.start.x) / line.dx())
147        };
148        let ty = if line.dy() == T::zero() {
149            None
150        } else {
151            Some((point.y() - line.start.y) / line.dy())
152        };
153        let contains = match (tx, ty) {
154            (None, None) => {
155                // Degenerate line
156                point.0 == line.start
157            }
158            (Some(t), None) => {
159                // Horizontal line
160                point.y() == line.start.y && T::zero() <= t && t <= T::one()
161            }
162            (None, Some(t)) => {
163                // Vertical line
164                point.x() == line.start.x && T::zero() <= t && t <= T::one()
165            }
166            (Some(t_x), Some(t_y)) => {
167                // All other lines
168                (t_x - t_y).abs() <= T::epsilon() && T::zero() <= t_x && t_x <= T::one()
169            }
170        };
171        if contains {
172            return true;
173        }
174    }
175    false
