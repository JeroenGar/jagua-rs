Convert (infalliby) the type of a geometry’s coordinate value.

§Examples

use geo::{Convert, LineString, line_string};

let line_string_32: LineString<f32> = line_string![
    (x: 5., y: 10.),
    (x: 3., y: 1.),
    (x: 8., y: 9.),
];

let line_string_64: LineString<f64> = line_string_32.convert();
